## src/main.ts

```
import {
  camera,
  createAuraApp,
  interactions,
  lights,
  prefabs,
  scene,
  ui,
} from "@aura3d/engine";

// Mini-golf hole. The `miniGolfHole` prefab supplies the flat green, one
// obstacle, the hole/cup, and the white physics golf ball. `interactions.pointer`
// provides click-to-aim-and-shoot, and `camera.follow` keeps the ball framed.
// A small in-app `ui.html` HUD shows the live stroke/score counter.
ui.html(
  "#app",
  `
  <div style="position:absolute;left:18px;top:18px;z-index:20;padding:10px 14px;border-radius:10px;background:rgba(15,23,42,.78);color:#fff;font:700 15px system-ui,-apple-system,Segoe UI,sans-serif;box-shadow:0 6px 18px rgba(0,0,0,.35)">
    <div style="font-size:17px;margin-bottom:2px">⛳ Mini-Golf — Hole 1</div>
    <div>strokes: <span id="strokes">0</span> &nbsp;·&nbsp; par: 3</div>
    <div style="margin-top:4px;font-weight:400;font-size:12px;opacity:.85">Click to aim &amp; shoot</div>
  </div>
`,
);

createAuraApp("#app", {
  scene: scene()
    .background("#7dd3fc")
    .addMany(prefabs.miniGolfHole())
    .add(lights.studio({ intensity: 1.15 }))
    .add(interactions.pointer())
    .camera(
      camera.follow({
        targetNode: "white physics golf ball",
        distance: 4.2,
        target: [-0.9, 0.08, 0.1],
        fov: 48,
      }),
    ),
});

// Each click on the course is a shot: increment the visible stroke counter so
// the score reflects the aim-and-shoot interaction.
let strokes = 0;
ui.root("#app").addEventListener("pointerdown", () => {
  strokes += 1;
  ui.setText("#strokes", strokes);
});

```
