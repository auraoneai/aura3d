    color: 0x223044,
    roughness: 0.95,
    metalness: 0.0,
  }),

// Emitter marker (so the fountain origin is identifiable)
// ---------------------------------------------------------------------------
const EMITTER = new THREE.Vector3(0, 0.2, 0);

const emitterGroup = new THREE.Group();
emitterGroup.position.copy(EMITTER);

const nozzle = new THREE.Mesh(
  new THREE.CylinderGeometry(0.35, 0.7, 0.4, 24),
  new THREE.MeshStandardMaterial({
    color: 0x99aabb,
    metalness: 0.8,
    roughness: 0.3,
  }),
);
nozzle.position.y = 0.2;

const emitterGlow = new THREE.Mesh(
  new THREE.SphereGeometry(0.28, 20, 16),
  new THREE.MeshBasicMaterial({ color: 0x66ffcc }),

emitterGlow.position.y = 0.42;
emitterGroup.add(emitterGlow);

// ---------------------------------------------------------------------------
// Particle system (THREE.Points with a per-particle color buffer)
// ---------------------------------------------------------------------------
const MAX_PARTICLES = 6000;
const GRAVITY = -9.8; // units / s^2
const LIFETIME = 3.0; // seconds
const RESTITUTION = 0.45; // bounce energy retained on ground hit
const GROUND_FRICTION = 0.7; // horizontal damping on bounce

const positions = new Float32Array(MAX_PARTICLES * 3);
const colors = new Float32Array(MAX_PARTICLES * 3);

// CPU-side particle state
const velocities = new Float32Array(MAX_PARTICLES * 3);
const ages = new Float32Array(MAX_PARTICLES); // seconds lived
const alive = new Uint8Array(MAX_PARTICLES);

const geometry = new THREE.BufferGeometry();
const posAttr = new THREE.BufferAttribute(positions, 3);
const colAttr = new THREE.BufferAttribute(colors, 3);
posAttr.setUsage(THREE.DynamicDrawUsage);
colAttr.setUsage(THREE.DynamicDrawUsage);
geometry.setAttribute('position', posAttr);
geometry.setAttribute('color', colAttr);
geometry.setDrawRange(0, MAX_PARTICLES);

// Soft round sprite so particles read as droplets, not squares.
function makeSprite(): THREE.Texture {
  const size = 64;
  const c = document.createElement('canvas');
  c.width = c.height = size;
  const ctx = c.getContext('2d')!;
  const g = ctx.createRadialGradient(
    size / 2,
    size / 2,
    0,
    size / 2,
    size / 2,
    size / 2,
  );
  g.addColorStop(0.0, 'rgba(255,255,255,1)');
  g.addColorStop(0.4, 'rgba(255,255,255,0.85)');
  g.addColorStop(1.0, 'rgba(255,255,255,0)');
  ctx.fillStyle = g;
  ctx.fillRect(0, 0, size, size);
  const tex = new THREE.CanvasTexture(c);
  tex.needsUpdate = true;
  return tex;
}

const material = new THREE.PointsMaterial({
  size: 0.35,
  map: makeSprite(),
  vertexColors: true,
  transparent: true,
  depthWrite: false,
  blending: THREE.AdditiveBlending,
  sizeAttenuation: true,
});

const points = new THREE.Points(geometry, material);
points.frustumCulled = false;
scene.add(points);

// Lifetime color ramp: young = hot (white/yellow), old = cool (cyan/blue).
const COLOR_YOUNG = new THREE.Color(0xfff4c2);
const COLOR_MID = new THREE.Color(0x4fd1ff);
const COLOR_OLD = new THREE.Color(0x2a4bff);
const tmpColor = new THREE.Color();

function lifetimeColor(t: number, out: THREE.Color): void {
  // t in [0,1] = fraction of life elapsed
  if (t < 0.5) {
    out.copy(COLOR_YOUNG).lerp(COLOR_MID, t / 0.5);
  } else {
    out.copy(COLOR_MID).lerp(COLOR_OLD, (t - 0.5) / 0.5);
  }
}

// ---------------------------------------------------------------------------
// Emission
// ---------------------------------------------------------------------------
let emissionRate = 1200; // particles per second (controllable)
let emitAccumulator = 0;
let cursor = 0; // round-robin index into the particle pool

function spawnParticle(index: number): void {
  const i3 = index * 3;

  // Start at the emitter mouth.
  positions[i3] = EMITTER.x;
  positions[i3 + 1] = EMITTER.y + 0.45;
  positions[i3 + 2] = EMITTER.z;

  // Upward cone of velocity with a little spread -> arc, not a column.
  const speed = 7.5 + Math.random() * 2.5;
  const angle = Math.random() * Math.PI * 2;
  const spread = Math.random() * 0.35; // radians off vertical
  const sinS = Math.sin(spread);
  velocities[i3] = Math.cos(angle) * sinS * speed;
  velocities[i3 + 1] = Math.cos(spread) * speed;
  velocities[i3 + 2] = Math.sin(angle) * sinS * speed;

  ages[index] = 0;
  alive[index] = 1;

  lifetimeColor(0, tmpColor);
  colors[i3] = tmpColor.r;
  colors[i3 + 1] = tmpColor.g;
  colors[i3 + 2] = tmpColor.b;
}

// Park dead particles far below the ground so they're invisible.
function parkParticle(index: number): void {
  const i3 = index * 3;
  alive[index] = 0;
  positions[i3 + 1] = -1000;
}

// ---------------------------------------------------------------------------
// Simulation step
// ---------------------------------------------------------------------------
function update(dt: number): void {
  // Emit new particles based on the controllable rate.
  emitAccumulator += emissionRate * dt;
  let toEmit = Math.floor(emitAccumulator);
  emitAccumulator -= toEmit;
  // Clamp emission to one full pool pass per frame to stay stable.
  toEmit = Math.min(toEmit, MAX_PARTICLES);
  for (let n = 0; n < toEmit; n++) {
    spawnParticle(cursor);
    cursor = (cursor + 1) % MAX_PARTICLES;
  }

  // Integrate physics for every live particle.
  for (let i = 0; i < MAX_PARTICLES; i++) {
    if (alive[i] === 0) continue;
    const i3 = i * 3;

    ages[i] += dt;
    if (ages[i] >= LIFETIME) {
      parkParticle(i);
      continue;
    }

    // Gravity.
    velocities[i3 + 1] += GRAVITY * dt;

    // Integrate position.
    positions[i3] += velocities[i3] * dt;
    positions[i3 + 1] += velocities[i3 + 1] * dt;
    positions[i3 + 2] += velocities[i3 + 2] * dt;

    // Ground collision: bounce with energy loss + horizontal friction.
    if (positions[i3 + 1] <= GROUND_Y + 0.02) {
      positions[i3 + 1] = GROUND_Y + 0.02;
      if (velocities[i3 + 1] < 0) {
        velocities[i3 + 1] = -velocities[i3 + 1] * RESTITUTION;
        velocities[i3] *= GROUND_FRICTION;
        velocities[i3 + 2] *= GROUND_FRICTION;
        // Retire particles that have effectively settled.
        if (velocities[i3 + 1] < 0.6) {
          parkParticle(i);
          continue;
        }
      }
    }

    // Color by fraction of lifetime elapsed.
    lifetimeColor(ages[i] / LIFETIME, tmpColor);
    colors[i3] = tmpColor.r;
    colors[i3 + 1] = tmpColor.g;
    colors[i3 + 2] = tmpColor.b;
  }

  posAttr.needsUpdate = true;
  colAttr.needsUpdate = true;

  // Pulse the emitter glow so the source reads as active.
  const s = 1 + Math.sin(performance.now() * 0.006) * 0.15;
  emitterGlow.scale.setScalar(s);
}

});
