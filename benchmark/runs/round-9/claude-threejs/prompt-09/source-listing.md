// Prompt 09: Animated Primitive Humanoid
// A humanoid figure assembled entirely from Three.js primitive geometries
// (sphere head, cylinder torso/neck, box limbs) performing a procedural
// walk-cycle animation as it strides across a ground plane.

import * as THREE from 'three';

// ---------------------------------------------------------------------------
// Renderer
// ---------------------------------------------------------------------------
const app = document.getElementById('app') as HTMLDivElement;

const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
renderer.setSize(window.innerWidth, window.innerHeight);
renderer.shadowMap.enabled = true;
renderer.shadowMap.type = THREE.PCFSoftShadowMap;
app.appendChild(renderer.domElement);

// Make the canvas fill the viewport.
Object.assign(document.body.style, { margin: '0', overflow: 'hidden' });
Object.assign(renderer.domElement.style, { display: 'block' });

// Scene & Camera
// ---------------------------------------------------------------------------
const scene = new THREE.Scene();
scene.background = new THREE.Color(0x101522);
scene.fog = new THREE.Fog(0x101522, 20, 48);

const camera = new THREE.PerspectiveCamera(
  50,
  window.innerWidth / window.innerHeight,
  0.1,
  100,
);
camera.position.set(6.5, 4.2, 11);
camera.lookAt(0, 1.5, 0);

// ---------------------------------------------------------------------------
// Lighting
// ---------------------------------------------------------------------------
scene.add(new THREE.HemisphereLight(0xbfd4ff, 0x202533, 0.9));

const sun = new THREE.DirectionalLight(0xfff2d9, 2.0);
sun.position.set(8, 14, 6);
sun.castShadow = true;
sun.shadow.mapSize.set(2048, 2048);
sun.shadow.camera.near = 1;
sun.shadow.camera.far = 50;
sun.shadow.camera.left = -16;
sun.shadow.camera.right = 16;
sun.shadow.camera.top = 16;
sun.shadow.camera.bottom = -16;
sun.shadow.bias = -0.0005;
scene.add(sun);

// ---------------------------------------------------------------------------
// Ground plane
// ---------------------------------------------------------------------------
const ground = new THREE.Mesh(
  new THREE.PlaneGeometry(80, 80),
  new THREE.MeshStandardMaterial({ color: 0x3a4256, roughness: 0.95, metalness: 0.0 }),
);
ground.rotation.x = -Math.PI / 2;
ground.receiveShadow = true;
scene.add(ground);

// A faint grid helps read the forward motion across the plane.
const grid = new THREE.GridHelper(80, 80, 0x5a6680, 0x2b3142);
(grid.material as THREE.Material).transparent = true;
(grid.material as THREE.Material).opacity = 0.35;
grid.position.y = 0.001;
scene.add(grid);

// ---------------------------------------------------------------------------
// Materials (shared)
// ---------------------------------------------------------------------------
const skinMat = new THREE.MeshStandardMaterial({ color: 0xe8b48c, roughness: 0.6 });
const shirtMat = new THREE.MeshStandardMaterial({ color: 0x2f7bd1, roughness: 0.7 });
const pantsMat = new THREE.MeshStandardMaterial({ color: 0x2b2f3a, roughness: 0.8 });
const limbMat = new THREE.MeshStandardMaterial({ color: 0x3a8be0, roughness: 0.7 });
const shoeMat = new THREE.MeshStandardMaterial({ color: 0x161922, roughness: 0.5 });

function mesh(geo: THREE.BufferGeometry, mat: THREE.Material): THREE.Mesh {
  const m = new THREE.Mesh(geo, mat);
  m.castShadow = true;
  m.receiveShadow = true;
  return m;
}

// ---------------------------------------------------------------------------
// Humanoid rig
// ---------------------------------------------------------------------------
// `walker` translates across the ground and bobs vertically.
// `body` holds the torso/head and the four limb pivots.
// Each limb is a Group whose pivot sits at the joint (shoulder / hip) so that
// rotating the group swings the limb like a pendulum.
const walker = new THREE.Group();
scene.add(walker);

const body = new THREE.Group();
walker.add(body);

// --- Torso (cylinder) ---
const torsoHeight = 1.4;
const torso = mesh(new THREE.CylinderGeometry(0.42, 0.5, torsoHeight, 20), shirtMat);
const torsoCenterY = 2.0; // hip joints sit at the bottom of the torso
torso.position.y = torsoCenterY;
body.add(torso);

// --- Neck (cylinder) + Head (sphere) ---
const neck = mesh(new THREE.CylinderGeometry(0.13, 0.13, 0.22, 12), skinMat);
neck.position.y = torsoCenterY + torsoHeight / 2 + 0.11;
body.add(neck);

const head = mesh(new THREE.SphereGeometry(0.42, 28, 24), skinMat);
head.position.y = neck.position.y + 0.11 + 0.36;
body.add(head);

// Simple face so the front of the figure reads clearly.
const eyeGeo = new THREE.SphereGeometry(0.06, 12, 12);
const eyeMat = new THREE.MeshStandardMaterial({ color: 0x14171f, roughness: 0.3 });
for (const dx of [-0.15, 0.15]) {
  const eye = mesh(eyeGeo, eyeMat);
  eye.position.set(dx, head.position.y + 0.05, 0.36);
  body.add(eye);
}

// --- Limb factory ---------------------------------------------------------
// Returns a pivot Group (placed at the joint) containing a limb built from a
// box "upper" segment and a box "lower" segment, with an optional foot.
// The lower segment lives in its own nested pivot so knees/elbows can flex.
interface Limb {
  pivot: THREE.Group; // shoulder / hip — primary swing
  lower: THREE.Group; // knee / elbow — secondary flex
}

function makeLimb(
  jointX: number,
  jointY: number,
  upperLen: number,
  lowerLen: number,
  thickness: number,
  mat: THREE.Material,
  withFoot: boolean,
): Limb {
  const pivot = new THREE.Group();
  pivot.position.set(jointX, jointY, 0);

  const upper = mesh(new THREE.BoxGeometry(thickness, upperLen, thickness), mat);
  upper.position.y = -upperLen / 2;
  pivot.add(upper);

  const lower = new THREE.Group();
  lower.position.y = -upperLen; // hinge at the end of the upper segment
  pivot.add(lower);

  const lowerSeg = mesh(
    new THREE.BoxGeometry(thickness * 0.9, lowerLen, thickness * 0.9),
    mat,
  );
  lowerSeg.position.y = -lowerLen / 2;
  lower.add(lowerSeg);

  if (withFoot) {
    const foot = mesh(new THREE.BoxGeometry(thickness * 1.1, 0.16, thickness * 1.9), shoeMat);
    foot.position.set(0, -lowerLen - 0.04, thickness * 0.45);
    lower.add(foot);
  } else {
    // hand
    const hand = mesh(new THREE.SphereGeometry(thickness * 0.62, 12, 12), skinMat);
    hand.position.y = -lowerLen - 0.02;
    lower.add(hand);
  }

  return { pivot, lower };
}

// Hips sit at the bottom of the torso.
const hipY = torsoCenterY - torsoHeight / 2;
const leftLeg = makeLimb(-0.22, hipY, 0.8, 0.8, 0.26, pantsMat, true);
const rightLeg = makeLimb(0.22, hipY, 0.8, 0.8, 0.26, pantsMat, true);
body.add(leftLeg.pivot, rightLeg.pivot);

// Shoulders sit near the top of the torso.
const shoulderY = torsoCenterY + torsoHeight / 2 - 0.18;
const leftArm = makeLimb(-0.6, shoulderY, 0.62, 0.6, 0.2, limbMat, false);
const rightArm = makeLimb(0.6, shoulderY, 0.62, 0.6, 0.2, limbMat, false);
body.add(leftArm.pivot, rightArm.pivot);

// ---------------------------------------------------------------------------
// Walk-cycle animation
// ---------------------------------------------------------------------------
const WALK_SPEED = 1.9; // metres / second across the ground
const STRIDE_HZ = 1.05; // stride cadence
const PATH_HALF = 11; // figure loops within [-PATH_HALF, +PATH_HALF] on X
// Start mid-stride so the very first rendered frame shows an animated pose
// (legs split, arms swinging) rather than a static neutral stance.
const PHASE_OFFSET = Math.PI * 0.55;

// The figure walks along +X and faces its direction of travel.
body.rotation.y = Math.PI / 2;
walker.position.x = -PATH_HALF;

const clock = new THREE.Clock();

function animate(): void {
  const t = clock.getElapsedTime();
  const phase = t * STRIDE_HZ * Math.PI * 2 + PHASE_OFFSET;

  // Forward travel with wrap-around so the figure keeps crossing the plane.
  walker.position.x = ((t * WALK_SPEED + PATH_HALF) % (PATH_HALF * 2)) - PATH_HALF;

  // Vertical bob: body rises twice per stride (once per footfall).
  walker.position.y = Math.abs(Math.sin(phase)) * 0.12;

  // Leg swing (opposite phase) + knee flex on the forward swing.
  const legSwing = 0.85;
  leftLeg.pivot.rotation.x = Math.sin(phase) * legSwing;
  rightLeg.pivot.rotation.x = Math.sin(phase + Math.PI) * legSwing;
  // Knees bend (always positive — knees only flex backward) as each leg lifts.
  leftLeg.lower.rotation.x = Math.max(0, -Math.sin(phase + 0.5)) * 1.1;
  rightLeg.lower.rotation.x = Math.max(0, -Math.sin(phase + Math.PI + 0.5)) * 1.1;

  // Arms swing opposite to the legs on the same side; elbows flex slightly.
  const armSwing = 0.7;
  leftArm.pivot.rotation.x = Math.sin(phase + Math.PI) * armSwing;
  rightArm.pivot.rotation.x = Math.sin(phase) * armSwing;
  leftArm.lower.rotation.x = (0.4 + 0.3 * Math.sin(phase + Math.PI)) * 0.9;
  rightArm.lower.rotation.x = (0.4 + 0.3 * Math.sin(phase)) * 0.9;

  // Subtle torso counter-rotation and bounce for a lively gait.
  body.rotation.z = Math.sin(phase) * 0.05;
  body.position.y = Math.sin(phase * 2) * 0.03;

  renderer.render(scene, camera);
}

renderer.setAnimationLoop(animate);

// ---------------------------------------------------------------------------
// Resize handling
// ---------------------------------------------------------------------------
window.addEventListener('resize', () => {
  camera.aspect = window.innerWidth / window.innerHeight;
  camera.updateProjectionMatrix();
  renderer.setSize(window.innerWidth, window.innerHeight);
});
