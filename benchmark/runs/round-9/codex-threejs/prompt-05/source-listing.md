function updateHover(clientX: number, clientY: number): void {
  const rect = renderer.domElement.getBoundingClientRect();
  pointer.x = ((clientX - rect.left) / rect.width) * 2 - 1;
  pointer.y = -(((clientY - rect.top) / rect.height) * 2 - 1);
}

renderer.domElement.addEventListener('pointermove', (event: PointerEvent) => {

});

function applyBarVisuals(bar: BarMesh, isHovered: boolean): void {
  bar.material.color.copy(heightColor(bar.userData.currentHeight, isHovered));
  bar.material.emissive.set(isHovered ? 0xffffff : 0x000000);
  bar.material.emissiveIntensity = isHovered ? 0.18 : 0;
  bar.scale.x = isHovered ? 1.14 : 1;
  bar.scale.z = isHovered ? 1.14 : 1;
}

function animateBars(elapsed: number): void {
  for (const bar of bars) {
    const duration = 1.35;
    if (elapsed >= bar.userData.retargetAt) {
      bar.userData.previousHeight = bar.userData.currentHeight;
      bar.userData.targetHeight = randomHeight();
      bar.userData.retargetAt = elapsed + 1.6 + Math.random() * 2.8;
    }

    const progress = THREE.MathUtils.clamp(1 - (bar.userData.retargetAt - elapsed) / duration, 0, 1);
    const eased = progress * progress * (3 - 2 * progress);
    const height = THREE.MathUtils.lerp(bar.userData.previousHeight, bar.userData.targetHeight, eased);
    bar.userData.currentHeight = height;
    bar.scale.y = height;
    bar.position.y = height / 2;
    applyBarVisuals(bar, bar === hoveredBar);
  }
}

function updateRaycast(): void {
  raycaster.setFromCamera(pointer, camera);
  const hit = raycaster.intersectObjects(bars, false)[0]?.object as BarMesh | undefined;

  if (hit !== hoveredBar) {
    hoveredBar = hit ?? null;
  }

  if (hoveredBar) {
    const { gridX, gridZ, currentHeight } = hoveredBar.userData;
    tooltip.textContent = `Column ${gridX}, Row ${gridZ}, Value ${currentHeight.toFixed(2)}`;
  } else {
    tooltip.textContent = 'Hover a bar';
  }
}

}
