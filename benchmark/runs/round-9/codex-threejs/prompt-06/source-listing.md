function setPointer(event: PointerEvent) {
  const rect = renderer.domElement.getBoundingClientRect();
  pointer.x = ((event.clientX - rect.left) / rect.width) * 2 - 1;
  pointer.y = -(((event.clientY - rect.top) / rect.height) * 2 - 1);

function updateAimVisual() {
  if (!aiming || sunk) {
    aimLine.visible = false;
    arrowHead.visible = false;
    aimPower = 0;
    updatePowerHud();
    return;

  const pullVector = ball.position.clone().sub(pointerWorld);
  pullVector.y = 0;
  const pullDistance = THREE.MathUtils.clamp(pullVector.length(), 0, 4.5);
  aimPower = pullDistance / 4.5;

  if (pullDistance > 0.05) {
    aimDirection.copy(pullVector.normalize());
  }

  const start = ball.position.clone().setY(0.16);
  const end = start.clone().add(aimDirection.clone().multiplyScalar(1.25 + aimPower * 4.2));

}

function collideWithRails() {
  const minX = -courseWidth / 2 + ballRadius;
  const maxX = courseWidth / 2 - ballRadius;
  const minZ = -courseLength / 2 + ballRadius;
  const maxZ = courseLength / 2 - ballRadius;

  if (ball.position.x < minX) {
    ball.position.x = minX;
    velocity.x = Math.abs(velocity.x) * 0.72;
  } else if (ball.position.x > maxX) {
    ball.position.x = maxX;
    velocity.x = -Math.abs(velocity.x) * 0.72;
  }

  if (ball.position.z < minZ) {
    ball.position.z = minZ;
    velocity.z = Math.abs(velocity.z) * 0.72;
  } else if (ball.position.z > maxZ) {
    ball.position.z = maxZ;
    velocity.z = -Math.abs(velocity.z) * 0.72;
  }
}

function collideWithObstacle() {
  const minX = obstacle.position.x - obstacle.halfWidth - ballRadius;
  const maxX = obstacle.position.x + obstacle.halfWidth + ballRadius;
  const minZ = obstacle.position.z - obstacle.halfDepth - ballRadius;
  const maxZ = obstacle.position.z + obstacle.halfDepth + ballRadius;

  if (ball.position.x < minX || ball.position.x > maxX || ball.position.z < minZ || ball.position.z > maxZ) {
    return;
  }

  const overlapLeft = Math.abs(ball.position.x - minX);
  const overlapRight = Math.abs(maxX - ball.position.x);
  const overlapNear = Math.abs(ball.position.z - minZ);
  const overlapFar = Math.abs(maxZ - ball.position.z);
  const smallest = Math.min(overlapLeft, overlapRight, overlapNear, overlapFar);

  if (smallest === overlapLeft) {
    ball.position.x = minX;
    velocity.x = -Math.abs(velocity.x) * 0.78;
  } else if (smallest === overlapRight) {
    ball.position.x = maxX;
    velocity.x = Math.abs(velocity.x) * 0.78;
  } else if (smallest === overlapNear) {
    ball.position.z = minZ;
    velocity.z = -Math.abs(velocity.z) * 0.78;
  } else {
    ball.position.z = maxZ;
    velocity.z = Math.abs(velocity.z) * 0.78;
  }
}

function checkHole() {
  if (sunk) {
    return;
  }

  const distanceToCup = ball.position.distanceTo(holePosition);
  const slowEnough = velocity.length() < 3.4;

  if (distanceToCup < 0.54 && slowEnough) {
    sunk = true;
    velocity.set(0, 0, 0);
    ball.position.set(holePosition.x, ballRadius * 0.34, holePosition.z);
    statusText.textContent = `Holed in ${strokes} ${strokes === 1 ? 'stroke' : 'strokes'}.`;
  }
}

function updatePhysics(delta: number) {
  if (sunk) {
    return;
  }

  previousBallPosition.copy(ball.position);
  ball.position.addScaledVector(velocity, delta);
  ball.position.y = ballRadius;

  collideWithRails();
  collideWithObstacle();
  checkHole();

  const speed = velocity.length();
  if (speed > 0.001) {
    const travel = ball.position.clone().sub(previousBallPosition);
    spinAxis.set(travel.z, 0, -travel.x).normalize();
    ball.rotateOnWorldAxis(spinAxis, travel.length() / ballRadius);
  }

  const friction = Math.pow(0.68, delta);
  velocity.multiplyScalar(friction);
  if (velocity.lengthSq() < 0.008 && !aiming) {
    velocity.set(0, 0, 0);
    if (!sunk) {
      statusText.textContent = 'Click and drag from the ball to aim.';
    }
  }
}

function updateCamera(delta: number) {
  const behindDirection = velocity.lengthSq() > 0.05 ? velocity.clone().normalize() : new THREE.Vector3(0, 0, -1);
  const desiredPosition = ball.position
    .clone()
    .add(new THREE.Vector3(-behindDirection.x * 5.4, 5.8, -behindDirection.z * 5.4 + 2.1));

  desiredPosition.x = THREE.MathUtils.clamp(desiredPosition.x, -7.8, 7.8);
  desiredPosition.z = THREE.MathUtils.clamp(desiredPosition.z, -2, 15);
  camera.position.lerp(desiredPosition, 1 - Math.pow(0.001, delta));

  cameraTarget.copy(ball.position);
  cameraTarget.y = 0.55;
  camera.lookAt(cameraTarget);

  ballMarker.position.set(ball.position.x, 0.055, ball.position.z);
}

function animate(now: number) {
  const delta = Math.min((now - lastTime) / 1000, 0.033);
  lastTime = now;

  updatePhysics(delta);
  updateAimVisual();
  updateCamera(delta);

  renderer.render(scene, camera);
  requestAnimationFrame(animate);
}

window.addEventListener('resize', () => {
  camera.aspect = window.innerWidth / window.innerHeight;
  camera.updateProjectionMatrix();
  renderer.setSize(window.innerWidth, window.innerHeight);
});

updatePowerHud();
requestAnimationFrame(animate);
