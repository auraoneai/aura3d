function applyRampCollision(body: Body): number {
  ramp.updateMatrixWorld();
  rampInverse.copy(ramp.matrixWorld).invert();
  rampNormal.set(0, 1, 0).transformDirection(ramp.matrixWorld).normalize();

  const insideZ = Math.abs(tempLocal.z) < rampSize.z / 2 + body.radius;
  const topPenetration = rampSize.y / 2 + body.radius - tempLocal.y;

  if (insideX && insideZ && topPenetration > 0 && tempLocal.y > -rampSize.y * 1.3) {
    tempWorld.copy(rampNormal).multiplyScalar(topPenetration + 0.004);

      body.velocity.addScaledVector(rampNormal, 0.05);

    body.velocity.add(slide);
    body.velocity.multiplyScalar(0.992);

}

  return 0;
}

function applyCubeCollisions(): number {

  for (let i = 0; i < bodies.length; i += 1) {

      const a = bodies[i];
      const b = bodies[j];
      const delta = tempWorld.subVectors(b.position, a.position);
      const distance = delta.length();
      const minimum = (a.size + b.size) * 0.55;

      if (distance > 0.0001 && distance < minimum) {
        const normal = delta.multiplyScalar(1 / distance);
        const penetration = minimum - distance;
        a.position.addScaledVector(normal, -penetration * 0.5);
        b.position.addScaledVector(normal, penetration * 0.5);

        const relativeSpeed = b.velocity.clone().sub(a.velocity).dot(normal);

          const impulse = -(1.0 + 0.32) * relativeSpeed * 0.5;
          a.velocity.addScaledVector(normal, -impulse);
          b.velocity.addScaledVector(normal, impulse);
        }

        contacts += 1;
      }
    }
  }

  return contacts;
}

function stepSimulation(deltaSeconds: number): void {

      body.position.addScaledVector(body.velocity, dt);

      contacts += applyFloorCollision(body);

      const horizontalLimit = 10.2;

      if (Math.abs(body.position.x) > horizontalLimit) {
        body.position.x = Math.sign(body.position.x) * horizontalLimit;
        body.velocity.x *= -0.42;

      if (Math.abs(body.position.z) > depthLimit) {

        body.velocity.z *= -0.42;

    body.mesh.rotation.x += body.angularVelocity.x * deltaSeconds;
    body.mesh.rotation.y += body.angularVelocity.y * deltaSeconds;
    body.mesh.rotation.z += body.angularVelocity.z * deltaSeconds;

}
