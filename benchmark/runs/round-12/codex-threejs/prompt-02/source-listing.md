function lifetimeColor(progress: number): THREE.Color {
  if (progress < 0.28) {
    return color.set(0xdffcff).lerp(new THREE.Color(0x38d9ff), progress / 0.28);

function spawnParticle(): void {
  if (activeParticles >= maxParticles) {
    return;
  }

  const index = activeParticles;
  const offset = index * 3;
  const angle = randomRange(0, Math.PI * 2);

}

function removeParticle(index: number): void {
  const last = activeParticles - 1;

    const targetOffset = index * 3;

    positions[targetOffset + 1] = positions[sourceOffset + 1];
    positions[targetOffset + 2] = positions[sourceOffset + 2];
    colors[targetOffset] = colors[sourceOffset];
    colors[targetOffset + 1] = colors[sourceOffset + 1];
    colors[targetOffset + 2] = colors[sourceOffset + 2];

    ages[index] = ages[last];
    lifetimes[index] = lifetimes[last];
    bounces[index] = bounces[last];

}

function updateParticles(delta: number): void {
  emissionAccumulator += emissionRate * delta;
  const spawnCount = Math.floor(emissionAccumulator);
  emissionAccumulator -= spawnCount;

  for (let i = 0; i < spawnCount; i += 1) {
    spawnParticle();
  }

  for (let i = activeParticles - 1; i >= 0; i -= 1) {
    ages[i] += delta;

    if (ages[i] >= lifetimes[i]) {
      removeParticle(i);
      continue;

    const offset = i * 3;
    velocities[i].addScaledVector(gravity, delta);
    positions[offset] += velocities[i].x * delta;
    positions[offset + 1] += velocities[i].y * delta;
    positions[offset + 2] += velocities[i].z * delta;

    if (positions[offset + 1] < 0.055) {
      positions[offset + 1] = 0.055;
      velocities[i].y *= -0.38;
      velocities[i].x *= 0.72;
      velocities[i].z *= 0.72;
      bounces[i] += 1;

        removeParticle(i);
        continue;
      }

    const particleColor = lifetimeColor(progress);
    colors[offset] = particleColor.r;
    colors[offset + 1] = particleColor.g;
    colors[offset + 2] = particleColor.b;
  }

  particleGeometry.setDrawRange(0, activeParticles);
  particleGeometry.attributes.position.needsUpdate = true;
  particleGeometry.attributes.color.needsUpdate = true;
}
