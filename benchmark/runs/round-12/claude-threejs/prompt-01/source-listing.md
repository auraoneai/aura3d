// Physics Playground — 50 falling cubes onto a tilted ramp.
//
// Raw Three.js (no external physics engine). A compact, self-contained
// rigid-body sim provides gravity, sphere-vs-OBB collisions against the ramp,
// ground and bounding walls, and sphere-vs-sphere collisions between cubes,
// all with restitution + friction. A live contact-count overlay and reset
// control are provided in the DOM, and OrbitControls drives the camera.

import * as THREE from 'three';

sun.shadow.camera.right = shadowExtent;
sun.shadow.camera.top = shadowExtent;

// ---------------------------------------------------------------------------
// Box colliders (oriented bounding boxes). Sphere-vs-OBB is evaluated in
// each box's local space, so any rotated slab works without special cases.
// ---------------------------------------------------------------------------

interface BoxCollider {
  half: THREE.Vector3;
  invMatrix: THREE.Matrix4;
  normalMatrix: THREE.Matrix3;
}

const colliders: BoxCollider[] = [];

function addBox(
  width: number,
  height: number,
  depth: number,
  material: THREE.Material,
  setup: (mesh: THREE.Mesh) => void,
  collidable = true,
): THREE.Mesh {
  const mesh = new THREE.Mesh(new THREE.BoxGeometry(width, height, depth), material);
  setup(mesh);
  scene.add(mesh);
  mesh.updateMatrixWorld(true);
  if (collidable) {
    colliders.push({
      half: new THREE.Vector3(width / 2, height / 2, depth / 2),
      invMatrix: new THREE.Matrix4().copy(mesh.matrixWorld).invert(),
      normalMatrix: new THREE.Matrix3().getNormalMatrix(mesh.matrixWorld),
    });
  }
  return mesh;
}

// Static geometry: ground, tilted ramp, containing walls
// ---------------------------------------------------------------------------

const ARENA = 38; // footprint width/depth of the play area

const ground = addBox(
  ARENA,
  2,
  ARENA,
  new THREE.MeshStandardMaterial({ color: 0x2a2f3a, roughness: 0.95 }),
  (m) => {
    m.position.set(0, -1, 0);
    m.receiveShadow = true;
  },

void ground;

const RAMP_W = 22; // length down the slope (local X)

  opacity: 0.22,

const wallT = 1;

// ---------------------------------------------------------------------------
// Dynamic cubes
// ---------------------------------------------------------------------------

const CUBE_COUNT = 50;
const CUBE_SIZE = 1.6;
const CUBE_R = CUBE_SIZE * 0.5; // inscribed collision-sphere radius

interface Body {
  mesh: THREE.Mesh;
  vel: THREE.Vector3;
  angVel: THREE.Vector3;
}

const palette = [0xff6b6b, 0xffd166, 0x06d6a0, 0x4cc9f0, 0xf72585, 0xb5179e, 0xff9f1c];
const cubeGeo = new THREE.BoxGeometry(CUBE_SIZE, CUBE_SIZE, CUBE_SIZE);
const bodies: Body[] = [];

for (let i = 0; i < CUBE_COUNT; i++) {
  const mesh = new THREE.Mesh(
    cubeGeo,
    new THREE.MeshStandardMaterial({
      color: palette[i % palette.length],
      roughness: 0.4,
      metalness: 0.15,
    }),
  );
  mesh.castShadow = true;
  mesh.receiveShadow = true;
  scene.add(mesh);
  bodies.push({ mesh, vel: new THREE.Vector3(), angVel: new THREE.Vector3() });
}

function resetBodies(): void {
  for (const b of bodies) {
    // Loose column above the high end of the ramp so cubes cascade down.
    b.mesh.position.set(
      -8 + Math.random() * 6,
      16 + Math.random() * 18,
      (Math.random() - 0.5) * (RAMP_D - 3),
    );
    b.mesh.quaternion
      .set(Math.random(), Math.random(), Math.random(), Math.random())
      .normalize();
    b.vel.set((Math.random() - 0.5) * 2, 0, (Math.random() - 0.5) * 2);
    b.angVel.set(
      (Math.random() - 0.5) * 3,
      (Math.random() - 0.5) * 3,
      (Math.random() - 0.5) * 3,
    );
  }
}

// ---------------------------------------------------------------------------
// Physics
// ---------------------------------------------------------------------------

const GRAVITY = -30;
const RESTITUTION = 0.28;
const FRICTION = 0.86; // tangential velocity retained on contact
const SUBSTEPS = 3;

let liveContacts = 0;

const _local = new THREE.Vector3();
const _closest = new THREE.Vector3();
const _delta = new THREE.Vector3();
const _normal = new THREE.Vector3();
const _tangent = new THREE.Vector3();

function resolveContact(
  b: Body,
  localNormal: THREE.Vector3,
  normalMatrix: THREE.Matrix3,
  penetration: number,
): void {
  _normal.copy(localNormal).applyMatrix3(normalMatrix).normalize();
  b.mesh.position.addScaledVector(_normal, penetration);

  const vn = b.vel.dot(_normal);
  if (vn < 0) {
    _tangent.copy(b.vel).addScaledVector(_normal, -vn); // tangential part
    b.vel.copy(_tangent).multiplyScalar(FRICTION);
    b.vel.addScaledVector(_normal, -vn * RESTITUTION);
    b.angVel.multiplyScalar(0.85);
  }
}

function collideWithColliders(b: Body): number {
  let hits = 0;
  const pos = b.mesh.position;
  for (const c of colliders) {
    _local.copy(pos).applyMatrix4(c.invMatrix);
    const h = c.half;

    const insideX = Math.abs(_local.x) <= h.x;
    const insideY = Math.abs(_local.y) <= h.y;
    const insideZ = Math.abs(_local.z) <= h.z;

    if (insideX && insideY && insideZ) {
      // Center inside the box — push out along the least-penetration axis.
      const px = h.x - Math.abs(_local.x);
      const py = h.y - Math.abs(_local.y);
      const pz = h.z - Math.abs(_local.z);
      _normal.set(0, 0, 0);
      let pen: number;
      if (px <= py && px <= pz) {
        _normal.x = Math.sign(_local.x) || 1;
        pen = px + CUBE_R;
      } else if (py <= pz) {
        _normal.y = Math.sign(_local.y) || 1;
        pen = py + CUBE_R;
      } else {
        _normal.z = Math.sign(_local.z) || 1;
        pen = pz + CUBE_R;
      }
      resolveContact(b, _normal, c.normalMatrix, pen);
      hits++;
      continue;
    }

    _closest.set(
      THREE.MathUtils.clamp(_local.x, -h.x, h.x),
      THREE.MathUtils.clamp(_local.y, -h.y, h.y),
      THREE.MathUtils.clamp(_local.z, -h.z, h.z),
    );
    _delta.copy(_local).sub(_closest);
    const distSq = _delta.lengthSq();
    if (distSq < CUBE_R * CUBE_R && distSq > 1e-9) {
      const dist = Math.sqrt(distSq);
      _normal.copy(_delta).divideScalar(dist);
      resolveContact(b, _normal, c.normalMatrix, CUBE_R - dist);
      hits++;
    }
  }
  return hits;
}

const _diff = new THREE.Vector3();
const _push = new THREE.Vector3();
const _relVel = new THREE.Vector3();

function collideCubes(): number {
  let pairs = 0;
  const minDist = CUBE_R * 2;
  const minDistSq = minDist * minDist;
  for (let i = 0; i < bodies.length; i++) {
    const a = bodies[i];
    for (let j = i + 1; j < bodies.length; j++) {
      const b = bodies[j];
      _diff.copy(b.mesh.position).sub(a.mesh.position);
      const dSq = _diff.lengthSq();
      if (dSq < minDistSq && dSq > 1e-6) {
        const d = Math.sqrt(dSq);
        _push.copy(_diff).divideScalar(d);
        const overlap = (minDist - d) * 0.5;
        a.mesh.position.addScaledVector(_push, -overlap);
        b.mesh.position.addScaledVector(_push, overlap);
        _relVel.copy(b.vel).sub(a.vel);
        const rn = _relVel.dot(_push);
        if (rn < 0) {
          const imp = -(1 + RESTITUTION) * rn * 0.5;
          a.vel.addScaledVector(_push, -imp);
          b.vel.addScaledVector(_push, imp);
        }
        pairs++;
      }
    }
  }
  return pairs;
}

const _spinAxis = new THREE.Vector3();
const _spinQuat = new THREE.Quaternion();

function step(dt: number): void {
  let contacts = 0;
  const h = dt / SUBSTEPS;
  for (let s = 0; s < SUBSTEPS; s++) {
    for (const b of bodies) {
      b.vel.y += GRAVITY * h;
      b.mesh.position.addScaledVector(b.vel, h);

      const angle = b.angVel.length() * h;
      if (angle > 1e-5) {
        _spinAxis.copy(b.angVel).normalize();
        _spinQuat.setFromAxisAngle(_spinAxis, angle);
        b.mesh.quaternion.premultiply(_spinQuat).normalize();
      }

      contacts += collideWithColliders(b);
    }
    contacts += collideCubes();
  }
  liveContacts = contacts;
}

function animate(): void {
  const dt = Math.min(clock.getDelta(), 1 / 30);
  step(dt);
  controls.update();

  fpsAccum += dt;
  fpsFrames++;
  if (fpsAccum >= 0.4) {
    fpsValue = Math.round(fpsFrames / fpsAccum);
    fpsAccum = 0;
    fpsFrames = 0;

}
