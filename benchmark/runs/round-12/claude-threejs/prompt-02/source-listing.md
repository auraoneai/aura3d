// Particle Fountain — Three.js
//
// Gravity-affected particles emitted upward from an emitter point, colored by
// lifetime, bouncing off a ground plane, with a live emission-rate control.

import * as THREE from 'three';
import { OrbitControls } from 'three/addons/controls/OrbitControls.js';

// ---------------------------------------------------------------------------
// Renderer / scene / camera
// ---------------------------------------------------------------------------

const app = document.getElementById('app') as HTMLDivElement;

// Full-viewport canvas with no default margins/scrollbars.
const reset = document.createElement('style');
reset.textContent =
  'html,body{margin:0;height:100%;overflow:hidden;background:#0a0e16;}' +
  '#app{position:fixed;inset:0;}canvas{display:block;}';
document.head.appendChild(reset);

const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
renderer.setSize(window.innerWidth, window.innerHeight);
renderer.setClearColor(0x0a0e16, 1);
app.appendChild(renderer.domElement);

const scene = new THREE.Scene();
scene.fog = new THREE.Fog(0x0a0e16, 30, 80);

const camera = new THREE.PerspectiveCamera(
  55,
  window.innerWidth / window.innerHeight,
  0.1,
  200,
);
camera.position.set(14, 12, 18);

const controls = new OrbitControls(camera, renderer.domElement);
controls.enableDamping = true;
controls.target.set(0, 4, 0);

// ---------------------------------------------------------------------------
// Lighting
// ---------------------------------------------------------------------------

scene.add(new THREE.AmbientLight(0xffffff, 0.5));

const keyLight = new THREE.DirectionalLight(0xffffff, 1.2);
keyLight.position.set(10, 20, 10);
scene.add(keyLight);

// ---------------------------------------------------------------------------
// Ground plane (particles collide against this)
// ---------------------------------------------------------------------------

const GROUND_Y = 0;

const ground = new THREE.Mesh(
  new THREE.PlaneGeometry(80, 80),
  new THREE.MeshStandardMaterial({
    color: 0x1b2435,
    roughness: 0.95,
    metalness: 0.0,
  }),
);
ground.rotation.x = -Math.PI / 2;
ground.position.y = GROUND_Y;
scene.add(ground);

const grid = new THREE.GridHelper(80, 40, 0x3a4a66, 0x222c3d);
grid.position.y = GROUND_Y + 0.01;
scene.add(grid);

// ---------------------------------------------------------------------------
// Emitter marker (identifiable source of the fountain)
// ---------------------------------------------------------------------------

const EMITTER = new THREE.Vector3(0, 0.4, 0);

const emitterGroup = new THREE.Group();
emitterGroup.position.copy(EMITTER);
scene.add(emitterGroup);

// A nozzle cone pointing up marks where particles originate.
const nozzle = new THREE.Mesh(
  new THREE.ConeGeometry(0.7, 1.4, 24, 1, true),
  new THREE.MeshStandardMaterial({
    color: 0x2c3650,
    roughness: 0.4,
    metalness: 0.6,
    side: THREE.DoubleSide,
  }),
);
nozzle.position.y = 0.3;
emitterGroup.add(nozzle);

// A glowing emissive core so the emitter reads clearly.
const emitterCore = new THREE.Mesh(
  new THREE.SphereGeometry(0.45, 24, 16),
  new THREE.MeshStandardMaterial({
    color: 0x66ddff,
    emissive: 0x33bbff,
    emissiveIntensity: 1.4,
    roughness: 0.3,
  }),
);
emitterCore.position.y = 0.9;
emitterGroup.add(emitterCore);

const emitterLight = new THREE.PointLight(0x66ddff, 6, 14, 2);
emitterLight.position.y = 1.1;
emitterGroup.add(emitterLight);

// ---------------------------------------------------------------------------
// Particle system
// ---------------------------------------------------------------------------

const MAX_PARTICLES = 6000;
const GRAVITY = -16; // units / s^2
const BOUNCE_DAMPING = 0.45; // velocity retained after hitting ground
const FRICTION = 0.7; // horizontal damping on bounce
const PARTICLE_RADIUS = 0.12; // collision offset above ground
const MIN_LIFE = 1.6;
const MAX_LIFE = 3.0;
const BASE_SPEED = 11; // initial upward speed
const SPREAD = 2.4; // horizontal velocity spread

// Per-particle parallel arrays. Live particles are kept packed in [0, live).
const px = new Float32Array(MAX_PARTICLES);
const py = new Float32Array(MAX_PARTICLES);
const pz = new Float32Array(MAX_PARTICLES);
const vx = new Float32Array(MAX_PARTICLES);
const vy = new Float32Array(MAX_PARTICLES);
const vz = new Float32Array(MAX_PARTICLES);
const age = new Float32Array(MAX_PARTICLES);
const life = new Float32Array(MAX_PARTICLES);

let live = 0;

// GPU attribute buffers.
const positions = new Float32Array(MAX_PARTICLES * 3);
const colors = new Float32Array(MAX_PARTICLES * 3);

const geometry = new THREE.BufferGeometry();
const positionAttr = new THREE.BufferAttribute(positions, 3);
const colorAttr = new THREE.BufferAttribute(colors, 3);
positionAttr.setUsage(THREE.DynamicDrawUsage);
colorAttr.setUsage(THREE.DynamicDrawUsage);
geometry.setAttribute('position', positionAttr);
geometry.setAttribute('color', colorAttr);
geometry.setDrawRange(0, 0);

// Soft round sprite so particles read as droplets, not square dots.
function makeSprite(): THREE.Texture {
  const size = 64;
  const canvas = document.createElement('canvas');
  canvas.width = canvas.height = size;
  const ctx = canvas.getContext('2d')!;
  const g = ctx.createRadialGradient(
    size / 2,
    size / 2,
    0,
    size / 2,
    size / 2,
    size / 2,
  );
  g.addColorStop(0.0, 'rgba(255,255,255,1)');
  g.addColorStop(0.3, 'rgba(255,255,255,0.85)');
  g.addColorStop(1.0, 'rgba(255,255,255,0)');
  ctx.fillStyle = g;
  ctx.fillRect(0, 0, size, size);
  const tex = new THREE.CanvasTexture(canvas);
  tex.colorSpace = THREE.SRGBColorSpace;
  return tex;
}

const material = new THREE.PointsMaterial({
  size: 0.5,
  map: makeSprite(),
  vertexColors: true,
  transparent: true,
  depthWrite: false,
  blending: THREE.AdditiveBlending,
  sizeAttenuation: true,
});

const points = new THREE.Points(geometry, material);
points.frustumCulled = false;
scene.add(points);

// Lifetime color ramp: young particles are hot/bright, fading through the
// spectrum as they age (clear, visible lifetime color variation).
const tmpColor = new THREE.Color();
function lifetimeColor(t: number, out: THREE.Color): void {
  // t in [0,1]; hue sweeps blue -> cyan -> green -> yellow -> red as it ages.
  const hue = THREE.MathUtils.lerp(0.58, 0.0, t);
  const lightness = THREE.MathUtils.lerp(0.7, 0.4, t);
  out.setHSL(hue, 1.0, lightness);
}

function spawnParticle(): void {
  if (live >= MAX_PARTICLES) return;
  const i = live++;

  px[i] = EMITTER.x + (Math.random() - 0.5) * 0.2;
  py[i] = EMITTER.y + 0.9;
  pz[i] = EMITTER.z + (Math.random() - 0.5) * 0.2;

  // Upward cone of velocity with a little randomized horizontal spread.
  const angle = Math.random() * Math.PI * 2;
  const radial = Math.random() * SPREAD;
  vx[i] = Math.cos(angle) * radial;
  vz[i] = Math.sin(angle) * radial;
  vy[i] = BASE_SPEED + (Math.random() - 0.5) * 2.0;

  age[i] = 0;
  life[i] = MIN_LIFE + Math.random() * (MAX_LIFE - MIN_LIFE);
}

function killParticle(i: number): void {
  // Swap the dead particle with the last live one to keep [0, live) packed.
  const last = live - 1;
  if (i !== last) {
    px[i] = px[last];
    py[i] = py[last];
    pz[i] = pz[last];
    vx[i] = vx[last];
    vy[i] = vy[last];
    vz[i] = vz[last];
    age[i] = age[last];
    life[i] = life[last];
  }
  live--;
}

// ---------------------------------------------------------------------------
// Emission-rate control (visible UI)
// ---------------------------------------------------------------------------

let emissionRate = 1400; // particles per second
let emitAccumulator = 0;

const ui = document.createElement('div');
ui.style.cssText = [
  'position:fixed',
  'top:16px',
  'left:16px',
  'padding:14px 16px',
  'background:rgba(12,18,30,0.82)',
  'border:1px solid rgba(120,180,255,0.35)',
  'border-radius:10px',
  'color:#dce8ff',
  'font:13px/1.4 system-ui,sans-serif',
  'user-select:none',
  'backdrop-filter:blur(4px)',
  'min-width:220px',
  'z-index:10',
].join(';');

const title = document.createElement('div');
title.textContent = 'Particle Fountain';
title.style.cssText = 'font-weight:600;font-size:15px;margin-bottom:8px;';
ui.appendChild(title);

const label = document.createElement('label');
label.style.cssText = 'display:block;margin-bottom:6px;';
const labelText = document.createElement('span');
const updateLabel = (): void => {
  labelText.textContent = `Emission rate: ${emissionRate} /s`;
};
updateLabel();
label.appendChild(labelText);
ui.appendChild(label);

const slider = document.createElement('input');
slider.type = 'range';
slider.min = '0';
slider.max = '4000';
slider.step = '50';
slider.value = String(emissionRate);
slider.style.cssText = 'width:100%;cursor:pointer;accent-color:#66ddff;';
slider.addEventListener('input', () => {
  emissionRate = Number(slider.value);
  updateLabel();
});
ui.appendChild(slider);

const hint = document.createElement('div');
hint.textContent = 'Drag to orbit • slider sets emission rate';
hint.style.cssText = 'margin-top:8px;opacity:0.6;font-size:11px;';
ui.appendChild(hint);

document.body.appendChild(ui);

// ---------------------------------------------------------------------------
// Simulation + render loop
// ---------------------------------------------------------------------------

function update(dt: number): void {
  // Emit new particles based on the controllable rate.
  emitAccumulator += emissionRate * dt;
  while (emitAccumulator >= 1) {
    spawnParticle();
    emitAccumulator -= 1;
  }

  // Integrate, collide, recycle.
  for (let i = 0; i < live; ) {
    age[i] += dt;
    if (age[i] >= life[i]) {
      killParticle(i);
      continue; // a swapped-in particle now occupies index i
    }

    vy[i] += GRAVITY * dt;

    px[i] += vx[i] * dt;
    py[i] += vy[i] * dt;
    pz[i] += vz[i] * dt;

    // Collide against the ground plane: bounce with energy loss.
    if (py[i] <= GROUND_Y + PARTICLE_RADIUS) {
      py[i] = GROUND_Y + PARTICLE_RADIUS;
      if (vy[i] < 0) {
        vy[i] = -vy[i] * BOUNCE_DAMPING;
        vx[i] *= FRICTION;
        vz[i] *= FRICTION;
      }
    }

    i++;
  }

  // Write packed live particles into the GPU buffers.
  for (let i = 0; i < live; i++) {
    const o = i * 3;
    positions[o] = px[i];
    positions[o + 1] = py[i];
    positions[o + 2] = pz[i];

    lifetimeColor(age[i] / life[i], tmpColor);
    colors[o] = tmpColor.r;
    colors[o + 1] = tmpColor.g;
    colors[o + 2] = tmpColor.b;
  }

  geometry.setDrawRange(0, live);
  positionAttr.addUpdateRange(0, live * 3);
  colorAttr.addUpdateRange(0, live * 3);
  positionAttr.needsUpdate = true;
  colorAttr.needsUpdate = true;

  // Gentle pulse on the emitter core so the source stays lively.
  const pulse = 1.2 + Math.sin(performance.now() * 0.004) * 0.3;
  emitterLight.intensity = pulse * 4;
}

const clock = new THREE.Clock();

function animate(): void {
  const dt = Math.min(clock.getDelta(), 0.05);
  update(dt);
  controls.update();
  renderer.render(scene, camera);
}

renderer.setAnimationLoop(animate);

// ---------------------------------------------------------------------------
// Resize
// ---------------------------------------------------------------------------

window.addEventListener('resize', () => {
  camera.aspect = window.innerWidth / window.innerHeight;
  camera.updateProjectionMatrix();
  renderer.setSize(window.innerWidth, window.innerHeight);
});
