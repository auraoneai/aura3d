## src/main.ts

```
// Procedural Solar System — Three.js
// Sun + 6 planets at distinct orbital distances/speeds, readable labels,
// bloom glow on the sun, and an orbit camera framing the whole system.

import * as THREE from 'three';
import { OrbitControls } from 'three/addons/controls/OrbitControls.js';
import { EffectComposer } from 'three/addons/postprocessing/EffectComposer.js';
import { RenderPass } from 'three/addons/postprocessing/RenderPass.js';
import { UnrealBloomPass } from 'three/addons/postprocessing/UnrealBloomPass.js';
import {
  CSS2DRenderer,
  CSS2DObject,
} from 'three/addons/renderers/CSS2DRenderer.js';

// ---------------------------------------------------------------------------
// Renderer + scene + camera
// ---------------------------------------------------------------------------

const app = document.getElementById('app') as HTMLDivElement;

const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
renderer.setSize(window.innerWidth, window.innerHeight);
renderer.toneMapping = THREE.ACESFilmicToneMapping;
renderer.toneMappingExposure = 1.0;
app.appendChild(renderer.domElement);

const scene = new THREE.Scene();
scene.background = new THREE.Color(0x05060d);

const camera = new THREE.PerspectiveCamera(
  55,
  window.innerWidth / window.innerHeight,
  0.1,
  2000,
);
// Angled, pulled-back view so the entire system is framed at startup.
camera.position.set(0, 42, 78);

// Label renderer overlays crisp HTML on top of the WebGL canvas so planet
// names stay readable and are unaffected by bloom.
const labelRenderer = new CSS2DRenderer();
labelRenderer.setSize(window.innerWidth, window.innerHeight);
labelRenderer.domElement.style.position = 'absolute';
labelRenderer.domElement.style.top = '0';
labelRenderer.domElement.style.left = '0';
labelRenderer.domElement.style.pointerEvents = 'none';
app.appendChild(labelRenderer.domElement);

const controls = new OrbitControls(camera, labelRenderer.domElement);
controls.enableDamping = true;
controls.dampingFactor = 0.06;
controls.minDistance = 20;
controls.maxDistance = 320;
controls.target.set(0, 0, 0);
controls.autoRotate = true;
controls.autoRotateSpeed = 0.35;
controls.update();

// ---------------------------------------------------------------------------
// Starfield backdrop
// ---------------------------------------------------------------------------

function makeStarfield(count: number): THREE.Points {
  const positions = new Float32Array(count * 3);
  for (let i = 0; i < count; i++) {
    // Distribute on a large sphere shell.
    const r = 600 + Math.random() * 600;
    const theta = Math.random() * Math.PI * 2;
    const phi = Math.acos(2 * Math.random() - 1);
    positions[i * 3 + 0] = r * Math.sin(phi) * Math.cos(theta);
    positions[i * 3 + 1] = r * Math.cos(phi);
    positions[i * 3 + 2] = r * Math.sin(phi) * Math.sin(theta);
  }
  const geo = new THREE.BufferGeometry();
  geo.setAttribute('position', new THREE.BufferAttribute(positions, 3));
  const mat = new THREE.PointsMaterial({
    color: 0xffffff,
    size: 1.4,
    sizeAttenuation: true,
    transparent: true,
    opacity: 0.85,
  });
  return new THREE.Points(geo, mat);
}

scene.add(makeStarfield(2500));

// ---------------------------------------------------------------------------
// Lights
// ---------------------------------------------------------------------------

// The sun is the primary light source for the planets.
const sunLight = new THREE.PointLight(0xfff2cc, 4000, 0, 2);
sunLight.position.set(0, 0, 0);
scene.add(sunLight);

// A faint ambient term keeps the night sides from going fully black.
scene.add(new THREE.AmbientLight(0x223044, 0.6));

// ---------------------------------------------------------------------------
// Label helper
// ---------------------------------------------------------------------------

function makeLabel(text: string): CSS2DObject {
  const div = document.createElement('div');
  div.textContent = text;
  div.style.color = '#ffffff';
  div.style.fontFamily =
    'system-ui, -apple-system, Segoe UI, Roboto, sans-serif';
  div.style.fontSize = '13px';
  div.style.fontWeight = '600';
  div.style.letterSpacing = '0.4px';
  div.style.padding = '2px 7px';
  div.style.borderRadius = '6px';
  div.style.background = 'rgba(8, 12, 24, 0.72)';
  div.style.border = '1px solid rgba(255, 255, 255, 0.22)';
  div.style.whiteSpace = 'nowrap';
  div.style.textShadow = '0 1px 3px rgba(0,0,0,0.9)';
  const label = new CSS2DObject(div);
  label.center.set(0.5, 0);
  return label;
}

// ---------------------------------------------------------------------------
// Sun
// ---------------------------------------------------------------------------

const sunRadius = 4;
const sun = new THREE.Mesh(
  new THREE.SphereGeometry(sunRadius, 64, 64),
  // Pure emissive material so it reads as a bright self-lit body and drives
  // the bloom pass (which is keyed on luminance threshold).
  new THREE.MeshBasicMaterial({ color: 0xffd34d }),
);
scene.add(sun);

// A soft additive halo around the sun reinforces the glow even before bloom.
const sunHalo = new THREE.Mesh(
  new THREE.SphereGeometry(sunRadius * 1.35, 32, 32),
  new THREE.MeshBasicMaterial({
    color: 0xffae33,
    transparent: true,
    opacity: 0.25,
    side: THREE.BackSide,
    blending: THREE.AdditiveBlending,
    depthWrite: false,
  }),
);
scene.add(sunHalo);

const sunLabel = makeLabel('Sun');
sunLabel.position.set(0, sunRadius + 1.2, 0);
sun.add(sunLabel);

// ---------------------------------------------------------------------------
// Planets
// ---------------------------------------------------------------------------

interface PlanetSpec {
  name: string;
  radius: number;
  distance: number; // orbital radius
  speed: number; // angular speed (radians/sec)
  color: number;
  spin: number; // self-rotation speed
}

const planetSpecs: PlanetSpec[] = [
  { name: 'Mercury', radius: 0.7, distance: 9, speed: 0.95, color: 0xb7a38a, spin: 0.6 },
  { name: 'Venus', radius: 1.1, distance: 14, speed: 0.68, color: 0xe7b96b, spin: 0.4 },
  { name: 'Earth', radius: 1.25, distance: 20, speed: 0.52, color: 0x3a7bd5, spin: 1.2 },
  { name: 'Mars', radius: 0.95, distance: 27, speed: 0.40, color: 0xd6552f, spin: 1.0 },
  { name: 'Jupiter', radius: 2.6, distance: 38, speed: 0.26, color: 0xc9a36b, spin: 1.8 },
  { name: 'Saturn', radius: 2.2, distance: 50, speed: 0.18, color: 0xe6d2a0, spin: 1.6 },
];

interface Planet {
  pivot: THREE.Group; // rotates around the sun
  mesh: THREE.Mesh;
  spec: PlanetSpec;
}

const planets: Planet[] = [];

// Faint orbit ring so the differing distances read clearly.
function makeOrbitRing(distance: number): THREE.Line {
  const segments = 160;
  const pts: THREE.Vector3[] = [];
  for (let i = 0; i <= segments; i++) {
    const a = (i / segments) * Math.PI * 2;
    pts.push(new THREE.Vector3(Math.cos(a) * distance, 0, Math.sin(a) * distance));
  }
  const geo = new THREE.BufferGeometry().setFromPoints(pts);
  const mat = new THREE.LineBasicMaterial({
    color: 0x4a5a78,
    transparent: true,
    opacity: 0.45,
  });
  return new THREE.Line(geo, mat);
}

for (const spec of planetSpecs) {
  scene.add(makeOrbitRing(spec.distance));

  const pivot = new THREE.Group();
  scene.add(pivot);

  const mesh = new THREE.Mesh(
    new THREE.SphereGeometry(spec.radius, 48, 48),
    new THREE.MeshStandardMaterial({
      color: spec.color,
      roughness: 0.85,
      metalness: 0.05,
    }),
  );
  // Spread planets out along their orbits so they don't start in a line.
  const startAngle = Math.random() * Math.PI * 2;
  mesh.position.set(
    Math.cos(startAngle) * spec.distance,
    0,
    Math.sin(startAngle) * spec.distance,
  );
  pivot.add(mesh);

  // Saturn gets a ring for flavor.
  if (spec.name === 'Saturn') {
    const ring = new THREE.Mesh(
      new THREE.RingGeometry(spec.radius * 1.4, spec.radius * 2.2, 64),
      new THREE.MeshBasicMaterial({
        color: 0xcbb789,
        side: THREE.DoubleSide,
        transparent: true,
        opacity: 0.7,
      }),
    );
    ring.rotation.x = Math.PI / 2.2;
    mesh.add(ring);
  }

  const label = makeLabel(spec.name);
  label.position.set(0, spec.radius + 0.9, 0);
  mesh.add(label);

  planets.push({ pivot, mesh, spec });
}

// ---------------------------------------------------------------------------
// Post-processing: bloom keyed to the bright sun
// ---------------------------------------------------------------------------

const composer = new EffectComposer(renderer);
composer.addPass(new RenderPass(scene, camera));

const bloomPass = new UnrealBloomPass(
  new THREE.Vector2(window.innerWidth, window.innerHeight),
  1.4, // strength
  0.55, // radius
  0.65, // threshold — only the bright sun/halo bloom, planets stay crisp
);
composer.addPass(bloomPass);

// ---------------------------------------------------------------------------
// Resize
// ---------------------------------------------------------------------------

window.addEventListener('resize', () => {
  const w = window.innerWidth;
  const h = window.innerHeight;
  camera.aspect = w / h;
  camera.updateProjectionMatrix();
  renderer.setSize(w, h);
  labelRenderer.setSize(w, h);
  composer.setSize(w, h);
});

// ---------------------------------------------------------------------------
// Animation loop
// ---------------------------------------------------------------------------

const clock = new THREE.Clock();

function animate(): void {
  const dt = clock.getDelta();
  const t = clock.elapsedTime;

  // Sun slow self-rotation + subtle halo pulse.
  sun.rotation.y += dt * 0.15;
  const pulse = 1 + Math.sin(t * 1.5) * 0.03;
  sunHalo.scale.setScalar(pulse);

  for (const p of planets) {
    p.pivot.rotation.y += dt * p.spec.speed;
    p.mesh.rotation.y += dt * p.spec.spin;
  }

  controls.update();
  composer.render();
  labelRenderer.render(scene, camera);
}

renderer.setAnimationLoop(animate);
```
