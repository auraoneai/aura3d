// Prompt 08: Procedurally generated city block
// Raw Three.js (v0.165.0). 20 box buildings of varying heights with windows,
// streets, street lights, and a day/night toggle that changes lighting + sky.

import * as THREE from 'three';
import { OrbitControls } from 'three/addons/controls/OrbitControls.js';

// ---------------------------------------------------------------------------
// Deterministic PRNG so the generated city is identical on every build/run.
// ---------------------------------------------------------------------------
function makeRng(seed: number): () => number {
  let s = seed >>> 0;
  return () => {
    // mulberry32
    s |= 0;
    s = (s + 0x6d2b79f5) | 0;
    let t = Math.imul(s ^ (s >>> 15), 1 | s);
    t = (t + Math.imul(t ^ (t >>> 7), 61 | t)) ^ t;
    return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
  };
}
const rng = makeRng(20250808);
const rand = (min: number, max: number) => min + rng() * (max - min);
const randInt = (min: number, max: number) => Math.floor(rand(min, max + 1));

// ---------------------------------------------------------------------------
// Renderer / scene / camera

const app = document.getElementById('app') as HTMLDivElement;

const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
renderer.setSize(window.innerWidth, window.innerHeight);
renderer.shadowMap.enabled = true;
renderer.shadowMap.type = THREE.PCFSoftShadowMap;
renderer.toneMapping = THREE.ACESFilmicToneMapping;
renderer.toneMappingExposure = 1.0;
app.appendChild(renderer.domElement);

  0.1,
  2000,

// ---------------------------------------------------------------------------
// City layout: 4 columns x 5 rows = 20 buildings, with streets between them.
// ---------------------------------------------------------------------------
const COLS = 4;
const ROWS = 5;
const CELL = 18; // center-to-center spacing of building lots
const LOT = 10; // sidewalk lot size (street width = CELL - LOT)
const groundY = 0;

const cityWidth = COLS * CELL;
const cityDepth = ROWS * CELL;

// ---------------------------------------------------------------------------
// Window facade textures. Each building gets a colour map (concrete wall +
// dark window grid) and an emissive map (windows that glow at night, some lit
// some dark). Sides use the facade; roof/floor use plain materials.
// ---------------------------------------------------------------------------
interface FacadeTextures {
  map: THREE.CanvasTexture;
  emissive: THREE.CanvasTexture;
}

function makeFacade(
  cols: number,
  floors: number,
  wall: string,
): FacadeTextures {
  const cell = 24; // px per window cell
  const w = cols * cell;
  const h = floors * cell;

  // --- colour map: wall with darker window panes + frames ---
  const cMap = document.createElement('canvas');
  cMap.width = w;
  cMap.height = h;
  const c = cMap.getContext('2d')!;
  c.fillStyle = wall;
  c.fillRect(0, 0, w, h);

  // --- emissive map: black wall, windows lit/unlit ---
  const cEmi = document.createElement('canvas');
  cEmi.width = w;
  cEmi.height = h;
  const e = cEmi.getContext('2d')!;
  e.fillStyle = '#000000';
  e.fillRect(0, 0, w, h);

  const pad = cell * 0.18; // window inset from cell edge
  const litColours = ['#fff4c2', '#ffe9a8', '#cfe8ff', '#fff8e0'];

  for (let fy = 0; fy < floors; fy++) {
    for (let fx = 0; fx < cols; fx++) {
      const x = fx * cell + pad;
      const y = fy * cell + pad;
      const ww = cell - pad * 2;
      const hh = cell - pad * 2;

      // dark glass pane on the colour map
      c.fillStyle = rng() > 0.5 ? '#1a2230' : '#222a36';
      c.fillRect(x, y, ww, hh);
      // subtle frame highlight
      c.strokeStyle = 'rgba(255,255,255,0.06)';
      c.lineWidth = 1;
      c.strokeRect(x + 0.5, y + 0.5, ww - 1, hh - 1);

      // emissive: ~55% of windows lit at night
      if (rng() < 0.55) {
        const col = litColours[randInt(0, litColours.length - 1)];
        e.fillStyle = col;
        e.fillRect(x, y, ww, hh);
      }
    }
  }

  const map = new THREE.CanvasTexture(cMap);
  const emissive = new THREE.CanvasTexture(cEmi);
  for (const t of [map, emissive]) {
    t.colorSpace = THREE.SRGBColorSpace;
    t.anisotropy = 4;
    t.magFilter = THREE.NearestFilter;
  }
  return { map, emissive };
}

// Ground (asphalt) — the streets read as the dark surface between lots.
// ---------------------------------------------------------------------------
const groundSize = Math.max(cityWidth, cityDepth) + CELL * 3;
const ground = new THREE.Mesh(
  new THREE.PlaneGeometry(groundSize, groundSize),
  new THREE.MeshStandardMaterial({ color: 0x2b2e33, roughness: 0.95 }),

// Lane markings (dashed yellow centre lines) along every street corridor.
const markMat = new THREE.MeshStandardMaterial({
  color: 0xffcc33,
  emissive: 0x000000,
  roughness: 0.7,
});
function laneLine(x: number, z: number, len: number, horizontal: boolean) {
  const dashLen = 2.2;
  const gap = 2.2;
  const count = Math.floor(len / (dashLen + gap));
  const start = -len / 2 + dashLen / 2;
  for (let i = 0; i < count; i++) {
    const offset = start + i * (dashLen + gap);
    const dash = new THREE.Mesh(
      new THREE.PlaneGeometry(horizontal ? dashLen : 0.4, horizontal ? 0.4 : dashLen),
      markMat,
    );
    dash.rotation.x = -Math.PI / 2;
    dash.position.set(horizontal ? x + offset : x, groundY + 0.02, horizontal ? z : z + offset);
    scene.add(dash);
  }
}
// vertical streets (run along Z) between columns
for (let i = 0; i <= COLS; i++) {
  const x = (i - COLS / 2) * CELL;
  laneLine(x, 0, cityDepth + CELL, false);
}
// horizontal streets (run along X) between rows
for (let j = 0; j <= ROWS; j++) {
  const z = (j - ROWS / 2) * CELL;
  laneLine(0, z, cityWidth + CELL, true);
}

// ---------------------------------------------------------------------------
// Buildings + sidewalks
// ---------------------------------------------------------------------------
const buildingMats: THREE.MeshStandardMaterial[] = []; // window emissive control
const sidewalkMat = new THREE.MeshStandardMaterial({ color: 0x6b6e73, roughness: 0.9 });
const roofMat = new THREE.MeshStandardMaterial({ color: 0x3a3d42, roughness: 0.85 });
const wallColours = ['#5b6470', '#6a6258', '#566070', '#736a64', '#4f5a63', '#69707a'];

const boxGeo = new THREE.BoxGeometry(1, 1, 1); // shared unit cube, scaled per building

for (let r = 0; r < ROWS; r++) {
  for (let col = 0; col < COLS; col++) {
    const cx = (col - (COLS - 1) / 2) * CELL;
    const cz = (r - (ROWS - 1) / 2) * CELL;

    // sidewalk slab under the lot
    const sidewalk = new THREE.Mesh(
      new THREE.BoxGeometry(LOT, 0.4, LOT),
      sidewalkMat,
    );
    sidewalk.position.set(cx, groundY + 0.2, cz);
    sidewalk.receiveShadow = true;
    scene.add(sidewalk);

    // building dimensions — heights vary widely
    const bw = rand(5.5, 7.5);
    const bd = rand(5.5, 7.5);
    const floors = randInt(3, 14); // varying heights
    const floorH = 3;
    const bh = floors * floorH;
    const winCols = Math.max(2, Math.round(bw / 2.2));

    const wall = wallColours[randInt(0, wallColours.length - 1)];
    const { map, emissive } = makeFacade(winCols, floors, wall);

    const facadeMat = new THREE.MeshStandardMaterial({
      map,
      emissiveMap: emissive,
      emissive: 0xffffff,
      emissiveIntensity: 0, // off during the day, raised at night
      roughness: 0.6,
      metalness: 0.05,
    });
    buildingMats.push(facadeMat);

    // material order: px, nx, py(top), ny(bottom), pz, nz
    const mats = [facadeMat, facadeMat, roofMat, roofMat, facadeMat, facadeMat];

    const building = new THREE.Mesh(boxGeo, mats);
    building.scale.set(bw, bh, bd);
    building.position.set(cx, groundY + 0.4 + bh / 2, cz);
    building.castShadow = true;
    building.receiveShadow = true;
    scene.add(building);
  }
}

interface StreetLamp {
  light: THREE.PointLight;
  head: THREE.MeshStandardMaterial;
}
const streetLamps: StreetLamp[] = [];
const poleMat = new THREE.MeshStandardMaterial({ color: 0x1c1f24, roughness: 0.7, metalness: 0.4 });
const poleGeo = new THREE.CylinderGeometry(0.18, 0.22, 6, 8);
const armGeo = new THREE.BoxGeometry(1.6, 0.15, 0.15);

function addStreetLight(x: number, z: number, faceX: number) {
  const group = new THREE.Group();

  const pole = new THREE.Mesh(poleGeo, poleMat);
  pole.position.y = 3;
  pole.castShadow = true;
  group.add(pole);

  const arm = new THREE.Mesh(armGeo, poleMat);
  arm.position.set(faceX * 0.8, 5.9, 0);
  group.add(arm);

  const headMat = new THREE.MeshStandardMaterial({
    color: 0x222222,
    emissive: 0xffd27f,
    emissiveIntensity: 0, // off during the day
  });
  const head = new THREE.Mesh(headGeo, headMat);
  head.position.set(faceX * 1.5, 5.8, 0);
  group.add(head);

  const light = new THREE.PointLight(0xffd27f, 0, 22, 1.6);
  light.position.set(faceX * 1.5, 5.6, 0);
  group.add(light);

  group.position.set(x, groundY + 0.4, z);
  scene.add(group);

  streetLamps.push({ light, head: headMat });
}

// place lamps along the vertical street corridors, alternating which side
for (let i = 0; i <= COLS; i++) {
  const x = (i - COLS / 2) * CELL;
  const faceX = i === 0 ? 1 : i === COLS ? -1 : (i % 2 === 0 ? 1 : -1);
  for (let r = 0; r < ROWS; r++) {
    const z = (r - (ROWS - 1) / 2) * CELL + CELL / 2;
    addStreetLight(x, z, faceX);
  }
}

const hemi = new THREE.HemisphereLight(0xbfd8ff, 0x404652, 1.0);
scene.add(hemi);

const sun = new THREE.DirectionalLight(0xfff2d6, 2.4);
sun.position.set(60, 90, 40);
sun.castShadow = true;
sun.shadow.mapSize.set(2048, 2048);
sun.shadow.camera.near = 1;
sun.shadow.camera.far = 400;
const sc = sun.shadow.camera as THREE.OrthographicCamera;
const span = Math.max(cityWidth, cityDepth);
sc.left = -span;
sc.right = span;
sc.top = span;
sc.bottom = -span;
sc.updateProjectionMatrix();
scene.add(sun);

const ambient = new THREE.AmbientLight(0x223044, 0); // boosted at night
scene.add(ambient);

// Sky: a large inward-facing sphere whose vertex colours we recolour on toggle.
const skyGeo = new THREE.SphereGeometry(800, 32, 16);
const skyMat = new THREE.ShaderMaterial({
  side: THREE.BackSide,
  uniforms: {
    topColor: { value: new THREE.Color(0x2f6fb0) },
    bottomColor: { value: new THREE.Color(0xbfd8ef) },
    offset: { value: 80 },
    exponent: { value: 0.7 },
  },
  vertexShader: /* glsl */ `
    varying vec3 vWorldPosition;
    void main() {
      vec4 wp = modelMatrix * vec4(position, 1.0);
      vWorldPosition = wp.xyz;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
    }
  `,
  fragmentShader: /* glsl */ `
    uniform vec3 topColor;
    uniform vec3 bottomColor;
    uniform float offset;
    uniform float exponent;
    varying vec3 vWorldPosition;
    void main() {
      float h = normalize(vWorldPosition + vec3(0.0, offset, 0.0)).y;
      float f = max(pow(max(h, 0.0), exponent), 0.0);
      gl_FragColor = vec4(mix(bottomColor, topColor, f), 1.0);
    }
  `,
});
const sky = new THREE.Mesh(skyGeo, skyMat);
scene.add(sky);

function applyState(s: typeof DAY) {
  scene.background = s.background;
  (skyMat.uniforms.topColor.value as THREE.Color).copy(s.skyTop);
  (skyMat.uniforms.bottomColor.value as THREE.Color).copy(s.skyBottom);
  (scene.fog as THREE.Fog).color.copy(s.fog);
  hemi.color.copy(s.hemiSky);
  hemi.groundColor.copy(s.hemiGround);

  stars.visible = s.starOpacity > 0;
