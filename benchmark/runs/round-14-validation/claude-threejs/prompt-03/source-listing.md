## src/main.ts

```
// Procedural Solar System
// Sun + 6 planets at different orbital distances and speeds, labelled planets,
// bloom on the sun, and an orbit camera that frames the whole system.

import * as THREE from 'three';
import { OrbitControls } from 'three/addons/controls/OrbitControls.js';
import { EffectComposer } from 'three/addons/postprocessing/EffectComposer.js';
import { RenderPass } from 'three/addons/postprocessing/RenderPass.js';
import { UnrealBloomPass } from 'three/addons/postprocessing/UnrealBloomPass.js';
import {
  CSS2DRenderer,
  CSS2DObject,
} from 'three/addons/renderers/CSS2DRenderer.js';

const app = document.querySelector<HTMLDivElement>('#app')!;
app.style.position = 'fixed';
app.style.inset = '0';
app.style.margin = '0';
app.style.overflow = 'hidden';
document.body.style.margin = '0';
document.body.style.background = '#000';

// --- Renderer -------------------------------------------------------------
const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
renderer.setSize(window.innerWidth, window.innerHeight);
renderer.toneMapping = THREE.ACESFilmicToneMapping;
renderer.toneMappingExposure = 1.0;
app.appendChild(renderer.domElement);

// Label renderer overlays the WebGL canvas; it must not capture pointer events
// so the orbit controls keep working.
const labelRenderer = new CSS2DRenderer();
labelRenderer.setSize(window.innerWidth, window.innerHeight);
labelRenderer.domElement.style.position = 'absolute';
labelRenderer.domElement.style.top = '0';
labelRenderer.domElement.style.left = '0';
labelRenderer.domElement.style.pointerEvents = 'none';
app.appendChild(labelRenderer.domElement);

// --- Scene & camera -------------------------------------------------------
const scene = new THREE.Scene();
scene.background = new THREE.Color('#01030a');

const camera = new THREE.PerspectiveCamera(
  55,
  window.innerWidth / window.innerHeight,
  0.1,
  2000,
);
// Positioned high and back so the entire system is framed.
camera.position.set(0, 55, 95);

const controls = new OrbitControls(camera, renderer.domElement);
controls.enableDamping = true;
controls.dampingFactor = 0.05;
controls.minDistance = 20;
controls.maxDistance = 400;
controls.target.set(0, 0, 0);

// --- Starfield backdrop ---------------------------------------------------
{
  const starCount = 2500;
  const positions = new Float32Array(starCount * 3);
  for (let i = 0; i < starCount; i++) {
    // Distribute on a large sphere shell around the system.
    const r = 400 + Math.random() * 600;
    const theta = Math.random() * Math.PI * 2;
    const phi = Math.acos(2 * Math.random() - 1);
    positions[i * 3] = r * Math.sin(phi) * Math.cos(theta);
    positions[i * 3 + 1] = r * Math.cos(phi);
    positions[i * 3 + 2] = r * Math.sin(phi) * Math.sin(theta);
  }
  const starGeo = new THREE.BufferGeometry();
  starGeo.setAttribute('position', new THREE.BufferAttribute(positions, 3));
  const starMat = new THREE.PointsMaterial({
    color: 0xffffff,
    size: 1.2,
    sizeAttenuation: true,
  });
  scene.add(new THREE.Points(starGeo, starMat));
}

// --- Lighting -------------------------------------------------------------
scene.add(new THREE.AmbientLight(0x222233, 1.0));
const sunLight = new THREE.PointLight(0xfff2cc, 4500, 0, 2);
scene.add(sunLight); // sits at the origin with the sun

// --- Sun ------------------------------------------------------------------
// MeshBasicMaterial keeps the sun fully bright (unlit) so it drives the bloom.
const sun = new THREE.Mesh(
  new THREE.SphereGeometry(6, 64, 64),
  new THREE.MeshBasicMaterial({ color: 0xffcc33 }),
);
scene.add(sun);

const sunLabel = makeLabel('Sun', '#ffd966');
sunLabel.position.set(0, 8.5, 0);
sun.add(sunLabel);

// --- Planets --------------------------------------------------------------
interface PlanetConfig {
  name: string;
  radius: number;
  distance: number;
  speed: number; // radians / second
  color: number;
  labelColor: string;
}

const planetConfigs: PlanetConfig[] = [
  { name: 'Mercury', radius: 0.9, distance: 12, speed: 0.95, color: 0xa9a9a9, labelColor: '#cfcfcf' },
  { name: 'Venus', radius: 1.5, distance: 17, speed: 0.72, color: 0xe6b873, labelColor: '#f0cf9a' },
  { name: 'Earth', radius: 1.7, distance: 23, speed: 0.55, color: 0x3a7bd5, labelColor: '#7db4ff' },
  { name: 'Mars', radius: 1.3, distance: 30, speed: 0.42, color: 0xc1440e, labelColor: '#ff8a5c' },
  { name: 'Jupiter', radius: 3.6, distance: 40, speed: 0.26, color: 0xd9a066, labelColor: '#f2c896' },
  { name: 'Saturn', radius: 3.0, distance: 52, speed: 0.18, color: 0xead6a4, labelColor: '#f4e6c4' },
];

interface Planet {
  pivot: THREE.Group;
  speed: number;
}

const planets: Planet[] = [];

for (const cfg of planetConfigs) {
  // Pivot at the origin; rotating it sweeps the planet around its orbit.
  const pivot = new THREE.Group();
  scene.add(pivot);

  const mesh = new THREE.Mesh(
    new THREE.SphereGeometry(cfg.radius, 48, 48),
    new THREE.MeshStandardMaterial({
      color: cfg.color,
      roughness: 0.85,
      metalness: 0.1,
    }),
  );
  mesh.position.set(cfg.distance, 0, 0);
  pivot.add(mesh);

  // Saturn gets a simple ring for extra readability.
  if (cfg.name === 'Saturn') {
    const ring = new THREE.Mesh(
      new THREE.RingGeometry(cfg.radius * 1.4, cfg.radius * 2.2, 64),
      new THREE.MeshBasicMaterial({
        color: 0xcbb683,
        side: THREE.DoubleSide,
        transparent: true,
        opacity: 0.7,
      }),
    );
    ring.rotation.x = -Math.PI / 2.2;
    mesh.add(ring);
  }

  // Visible orbit ring on the ecliptic plane.
  const orbit = new THREE.Mesh(
    new THREE.RingGeometry(cfg.distance - 0.05, cfg.distance + 0.05, 128),
    new THREE.MeshBasicMaterial({
      color: 0x335577,
      side: THREE.DoubleSide,
      transparent: true,
      opacity: 0.4,
    }),
  );
  orbit.rotation.x = -Math.PI / 2;
  scene.add(orbit);

  // Label floats above the planet and is parented to it so it tracks the orbit.
  const label = makeLabel(cfg.name, cfg.labelColor);
  label.position.set(0, cfg.radius + 1.8, 0);
  mesh.add(label);

  // Give a non-zero random-ish starting angle so planets are spread out.
  pivot.rotation.y = cfg.distance * 0.7;

  planets.push({ pivot, speed: cfg.speed });
}

// --- Post-processing: bloom on the sun ------------------------------------
const composer = new EffectComposer(renderer);
composer.addPass(new RenderPass(scene, camera));
const bloomPass = new UnrealBloomPass(
  new THREE.Vector2(window.innerWidth, window.innerHeight),
  1.4, // strength
  0.7, // radius
  0.85, // threshold — only the bright sun passes
);
composer.addPass(bloomPass);

// --- Resize ---------------------------------------------------------------
window.addEventListener('resize', () => {
  const w = window.innerWidth;
  const h = window.innerHeight;
  camera.aspect = w / h;
  camera.updateProjectionMatrix();
  renderer.setSize(w, h);
  composer.setSize(w, h);
  labelRenderer.setSize(w, h);
});

// --- Animation loop -------------------------------------------------------
const clock = new THREE.Clock();

function animate(): void {
  const dt = clock.getDelta();

  sun.rotation.y += dt * 0.1;
  for (const planet of planets) {
    planet.pivot.rotation.y += planet.speed * dt;
  }

  controls.update();
  composer.render();
  labelRenderer.render(scene, camera);
}

renderer.setAnimationLoop(animate);

// --- Helpers --------------------------------------------------------------
function makeLabel(text: string, color: string): CSS2DObject {
  const el = document.createElement('div');
  el.textContent = text;
  el.style.color = color;
  el.style.fontFamily =
    'system-ui, -apple-system, Segoe UI, Roboto, sans-serif';
  el.style.fontSize = '13px';
  el.style.fontWeight = '600';
  el.style.letterSpacing = '0.04em';
  el.style.padding = '2px 7px';
  el.style.borderRadius = '6px';
  el.style.background = 'rgba(0, 0, 0, 0.55)';
  el.style.border = '1px solid rgba(255, 255, 255, 0.18)';
  el.style.whiteSpace = 'nowrap';
  el.style.textShadow = '0 1px 3px rgba(0, 0, 0, 0.9)';
  return new CSS2DObject(el);
}
```
