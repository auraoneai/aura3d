// Particle Fountain — raw Three.js
// Gravity-affected particles emitted upward from a point, colored by lifetime,
// colliding against a ground plane, with a controllable emission rate.

import * as THREE from 'three';
import { OrbitControls } from 'three/addons/controls/OrbitControls.js';

// ---------------------------------------------------------------------------
// Renderer / scene / camera
// ---------------------------------------------------------------------------
const app = document.getElementById('app') as HTMLDivElement;

const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
renderer.setSize(window.innerWidth, window.innerHeight);
renderer.setClearColor(0x0a0e18, 1);
app.appendChild(renderer.domElement);

const scene = new THREE.Scene();
scene.fog = new THREE.Fog(0x0a0e18, 18, 48);

const camera = new THREE.PerspectiveCamera(
  55,
  window.innerWidth / window.innerHeight,
  0.1,
  200,
);
camera.position.set(9, 7, 12);

const controls = new OrbitControls(camera, renderer.domElement);
controls.enableDamping = true;
controls.target.set(0, 2.5, 0);

// ---------------------------------------------------------------------------
// Lighting
// ---------------------------------------------------------------------------
scene.add(new THREE.HemisphereLight(0xbfd4ff, 0x202028, 0.9));
const keyLight = new THREE.DirectionalLight(0xffffff, 1.1);
keyLight.position.set(6, 12, 8);
scene.add(keyLight);

// ---------------------------------------------------------------------------
// Ground plane
// ---------------------------------------------------------------------------
const GROUND_Y = 0;
const ground = new THREE.Mesh(
  new THREE.PlaneGeometry(60, 60),
  new THREE.MeshStandardMaterial({
    color: 0x2a3240,
    roughness: 0.95,
    metalness: 0.0,
  }),
);
ground.rotation.x = -Math.PI / 2;
ground.position.y = GROUND_Y;
scene.add(ground);

const grid = new THREE.GridHelper(60, 60, 0x4a5870, 0x2c3444);
(grid.material as THREE.Material).transparent = true;
(grid.material as THREE.Material).opacity = 0.5;
grid.position.y = GROUND_Y + 0.001;
scene.add(grid);

// ---------------------------------------------------------------------------
// Emitter (identifiable marker at the emission point)
// ---------------------------------------------------------------------------
const EMITTER = new THREE.Vector3(0, 0.15, 0);

const emitterGroup = new THREE.Group();
emitterGroup.position.copy(EMITTER);

// A nozzle cone pointing upward.
const nozzle = new THREE.Mesh(
  new THREE.ConeGeometry(0.45, 0.9, 24, 1, true),
  new THREE.MeshStandardMaterial({
    color: 0x9bb0c8,
    metalness: 0.7,
    roughness: 0.3,
    side: THREE.DoubleSide,
  }),
);
nozzle.position.y = 0.45;
emitterGroup.add(nozzle);

// A glowing emission core so the source point is obvious.
const core = new THREE.Mesh(
  new THREE.SphereGeometry(0.22, 24, 16),
  new THREE.MeshBasicMaterial({ color: 0x66ddff }),
);
core.position.y = 0.9;
emitterGroup.add(core);

const coreGlow = new THREE.PointLight(0x66ddff, 6, 8, 2);
coreGlow.position.y = 0.9;
emitterGroup.add(coreGlow);

scene.add(emitterGroup);

// ---------------------------------------------------------------------------
// Particle system (pooled, simulated in JS)
// ---------------------------------------------------------------------------
const MAX_PARTICLES = 6000;
const GRAVITY = -9.81;
const RESTITUTION = 0.45; // bounciness on ground collision
const PARTICLE_RADIUS = 0.04;
const LIFE_MIN = 2.4;
const LIFE_MAX = 3.6;

// Per-particle simulation state (compacted: indices [0, liveCount) are alive).
const px = new Float32Array(MAX_PARTICLES);
const py = new Float32Array(MAX_PARTICLES);
const pz = new Float32Array(MAX_PARTICLES);
const vx = new Float32Array(MAX_PARTICLES);
const vy = new Float32Array(MAX_PARTICLES);
const vz = new Float32Array(MAX_PARTICLES);
const age = new Float32Array(MAX_PARTICLES);
const life = new Float32Array(MAX_PARTICLES);

let liveCount = 0;

// GPU buffers fed each frame from the live prefix.
const positions = new Float32Array(MAX_PARTICLES * 3);
const colors = new Float32Array(MAX_PARTICLES * 3);

const geometry = new THREE.BufferGeometry();
const posAttr = new THREE.BufferAttribute(positions, 3);
const colAttr = new THREE.BufferAttribute(colors, 3);
posAttr.setUsage(THREE.DynamicDrawUsage);
colAttr.setUsage(THREE.DynamicDrawUsage);
geometry.setAttribute('position', posAttr);
geometry.setAttribute('color', colAttr);
geometry.setDrawRange(0, 0);

// Soft round sprite so particles read as droplets rather than squares.
const sprite = makeSpriteTexture();
const material = new THREE.PointsMaterial({
  size: 0.34,
  map: sprite,
  alphaTest: 0.2,
  transparent: true,
  depthWrite: false,
  vertexColors: true,
  blending: THREE.AdditiveBlending,
  sizeAttenuation: true,
});

const points = new THREE.Points(geometry, material);
points.frustumCulled = false;
scene.add(points);

// Lifetime color ramp: young = hot cyan-white, old = deep orange/red.
const youngColor = new THREE.Color(0x9ff5ff);
const midColor = new THREE.Color(0x4f7bff);
const oldColor = new THREE.Color(0xff5a2a);
const tmpColor = new THREE.Color();

function colorByLife(t: number, out: THREE.Color): void {
  // t in [0,1]: 0 = just born, 1 = about to die.
  if (t < 0.5) {
    out.copy(youngColor).lerp(midColor, t / 0.5);
  } else {
    out.copy(midColor).lerp(oldColor, (t - 0.5) / 0.5);
  }
}

function spawnParticle(): void {
  if (liveCount >= MAX_PARTICLES) return;
  const i = liveCount++;

  px[i] = EMITTER.x;
  py[i] = EMITTER.y + 0.9; // emit from the glowing core
  pz[i] = EMITTER.z;

  // Upward cone of velocity: strong vertical, small radial spread.
  const speed = 6.2 + Math.random() * 2.2;
  const spreadAngle = Math.random() * 0.32; // radians from vertical
  const azimuth = Math.random() * Math.PI * 2;
  const sinA = Math.sin(spreadAngle);
  vx[i] = Math.cos(azimuth) * sinA * speed;
  vz[i] = Math.sin(azimuth) * sinA * speed;
  vy[i] = Math.cos(spreadAngle) * speed;

  age[i] = 0;
  life[i] = LIFE_MIN + Math.random() * (LIFE_MAX - LIFE_MIN);
}

function killParticle(i: number): void {
  // Swap-remove with the last live particle to keep [0, liveCount) compact.
  const last = liveCount - 1;
  if (i !== last) {
    px[i] = px[last]; py[i] = py[last]; pz[i] = pz[last];
    vx[i] = vx[last]; vy[i] = vy[last]; vz[i] = vz[last];
    age[i] = age[last]; life[i] = life[last];
  }
  liveCount--;
}

let emitAccumulator = 0;

function updateParticles(dt: number): void {
  // Emit according to the controllable rate.
  emitAccumulator += emissionRate * dt;
  let toSpawn = Math.floor(emitAccumulator);
  emitAccumulator -= toSpawn;
  while (toSpawn-- > 0) spawnParticle();

  // Integrate motion + ground collision.
  for (let i = 0; i < liveCount; ) {
    age[i] += dt;
    if (age[i] >= life[i]) {
      killParticle(i);
      continue; // re-process the swapped-in particle at this index
    }

    vy[i] += GRAVITY * dt;
    px[i] += vx[i] * dt;
    py[i] += vy[i] * dt;
    pz[i] += vz[i] * dt;

    // Collide against the ground plane.
    if (py[i] <= GROUND_Y + PARTICLE_RADIUS) {
      py[i] = GROUND_Y + PARTICLE_RADIUS;
      if (vy[i] < 0) vy[i] = -vy[i] * RESTITUTION;
      vx[i] *= 0.7; // friction
      vz[i] *= 0.7;
    }

    const t = age[i] / life[i];
    colorByLife(t, tmpColor);

    const o = i * 3;
    positions[o] = px[i];
    positions[o + 1] = py[i];
    positions[o + 2] = pz[i];
    colors[o] = tmpColor.r;
    colors[o + 1] = tmpColor.g;
    colors[o + 2] = tmpColor.b;

    i++;
  }

  geometry.setDrawRange(0, liveCount);
  posAttr.needsUpdate = true;
  colAttr.needsUpdate = true;
}

function makeSpriteTexture(): THREE.Texture {
  const size = 64;
  const canvas = document.createElement('canvas');
  canvas.width = canvas.height = size;
  const ctx = canvas.getContext('2d')!;
  const g = ctx.createRadialGradient(
    size / 2, size / 2, 0,
    size / 2, size / 2, size / 2,
  );
  g.addColorStop(0.0, 'rgba(255,255,255,1)');
  g.addColorStop(0.3, 'rgba(255,255,255,0.85)');
  g.addColorStop(1.0, 'rgba(255,255,255,0)');
  ctx.fillStyle = g;
  ctx.fillRect(0, 0, size, size);
  const tex = new THREE.CanvasTexture(canvas);
  tex.colorSpace = THREE.SRGBColorSpace;
  return tex;
}

}
slider.addEventListener('input', () => {

});
refreshLabel();

document.body.appendChild(ui);

window.addEventListener('resize', () => {
  camera.aspect = window.innerWidth / window.innerHeight;
  camera.updateProjectionMatrix();
  renderer.setSize(window.innerWidth, window.innerHeight);
});

const clock = new THREE.Clock();
let pulse = 0;

renderer.setAnimationLoop(() => {
  const dt = Math.min(clock.getDelta(), 0.05); // clamp to stay stable
  updateParticles(dt);

  // Subtle pulse of the emitter core for life.
  pulse += dt * 4;
  const s = 1 + Math.sin(pulse) * 0.08;
  core.scale.setScalar(s);
  coreGlow.intensity = 5 + Math.sin(pulse) * 1.5;

  readout.textContent = `Live particles: ${liveCount.toLocaleString()}`;

  controls.update();
  renderer.render(scene, camera);
});
