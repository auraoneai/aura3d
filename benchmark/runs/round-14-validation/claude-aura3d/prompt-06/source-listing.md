// Prompt 06: Mini-Golf Hole
// Flat green + one obstacle + physics ball + click-to-aim-and-shoot + score
// counter + follow camera on the ball.
//
// Built from the benchmark "06 Mini Golf" recipe shape using only public
// `@aura3d/engine` helpers confirmed in the shipped engine type surface:
//   - prefabs.miniGolfHole()  -> green, obstacle, white physics golf ball,
//                                 aim line, shot-power meter, ball trail,
//                                 contact flash, cup, score geometry
//   - camera.follow({ targetNode }) -> follow camera locked on the ball
//   - interactions.pointer()  -> click-to-aim-and-shoot pointer interaction
//   - ui.* (html/onClick/onInput/setText) -> visible strokes HUD + power meter
import {
  camera,
  createAuraApp,
  interactions,
  lights,
  prefabs,
  scene,
  timeline,
  ui
} from "@aura3d/engine";

// Name of the follow target inside the mini-golf prefab (per api-surface docs).
const BALL_NODE = "white physics golf ball";

// HUD: score counter (strokes) + aim/shot power meter + shoot button.
ui.html(
  "#app",
  `
  <div style="position:absolute;left:18px;top:18px;z-index:20;padding:8px 12px;border-radius:8px;background:rgba(15,23,42,.78);color:#fff;font:700 14px system-ui">
    strokes: <span id="strokes">0</span>
    &nbsp;|&nbsp; power
    <input id="power" type="range" min="0" max="100" value="56" />
    <span id="power-value">56</span>
  </div>
  <button id="shoot" type="button"
    style="position:absolute;left:18px;top:58px;z-index:20;padding:8px 12px;border-radius:8px;border:0;cursor:pointer;background:rgba(8,47,73,.85);color:#fff;font:700 14px system-ui">
    click to aim &amp; shoot
  </button>
`
);

// One app, one scene: flat green, single obstacle, physics ball, follow camera.
createAuraApp("#app", {
  scene: scene()
    .background("#7dd3fc")
    .addMany(prefabs.miniGolfHole())
    .add(lights.studio({ intensity: 1.15 }))
    .add(interactions.pointer())
    .camera(camera.follow({ targetNode: BALL_NODE, distance: 4.2 }))
    .timeline(timeline.loop({ seconds: 5 }))
});

// HUD wiring: live power readout + stroke (score) counter on each shot.
let strokes = 0;
ui.onInput("#power", (input) => ui.setText("#power-value", input.value));
ui.onClick("#shoot", () => {
  strokes += 1;
  ui.setText("#strokes", String(strokes));
});
