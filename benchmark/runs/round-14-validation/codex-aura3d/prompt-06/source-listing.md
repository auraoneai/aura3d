import {
  camera,
  createAuraApp,
  interactions,
  lights,
  prefabs,
  scene,
  timeline,
  ui
} from "@aura3d/engine";

ui.html("#app", `
  <div style="position:absolute;left:18px;top:18px;z-index:20;padding:9px 11px;border-radius:8px;background:rgba(15,23,42,.78);color:white;font:700 14px system-ui;box-shadow:0 8px 24px rgba(15,23,42,.22)">
    strokes: <span id="strokes">0</span> | power
    <input id="power" type="range" min="0" max="100" value="58" style="vertical-align:middle" />
    <span id="power-value">58</span>
  </div>
  <button id="shoot" type="button" style="position:absolute;left:18px;top:62px;z-index:20;padding:9px 11px;border:0;border-radius:8px;background:rgba(8,47,73,.88);color:white;font:700 14px system-ui;box-shadow:0 8px 24px rgba(8,47,73,.24);cursor:pointer">
    aim and shoot
  </button>
  <div id="shot-note" style="position:absolute;left:18px;top:108px;z-index:20;padding:7px 10px;border-radius:8px;background:rgba(20,83,45,.82);color:white;font:700 13px system-ui">
    click the button to aim down the green
  </div>
`);

let strokes = 0;

function buildScene() {
  return scene()
    .background("#7dd3fc")
    .addMany(prefabs.miniGolfHole())
    .add(lights.studio({ intensity: 1.15 }))
    .add(interactions.pointer({ target: "white physics golf ball" }))
    .camera(camera.follow({
      targetNode: "white physics golf ball",
      position: [2.8, 2.3, 4.6],
      target: [0, 0.18, 0],
      distance: 4.4,
      fov: 48
    }))
    .timeline(timeline.loop({ seconds: 5 }));
}

let app = createAuraApp("#app", {
  scene: buildScene()
});

ui.onInput("#power", (input) => {
  ui.setText("#power-value", input.value);
});

ui.onClick("#shoot", () => {
  strokes += 1;
  const power = Number(ui.range("#power").value);
  ui.setText("#strokes", strokes);
  ui.setText("#shot-note", `shot ${strokes}: aim line active, power ${power}`);

  app.dispose();
  app = createAuraApp("#app", {
    scene: buildScene()
  });
});
