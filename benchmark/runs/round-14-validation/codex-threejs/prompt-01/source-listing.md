function resolveRampContact(body: Body): number {
  const offset = body.mesh.position.clone().sub(rampPlanePoint);
  const x = offset.dot(rampTangent);
  const z = offset.dot(rampBinormal);
  const distance = offset.dot(rampNormal);
  const onRamp = Math.abs(x) <= rampHalf.x + body.radius && Math.abs(z) <= rampHalf.y + body.radius;

  if (!onRamp || distance >= body.radius) {
    return 0;
  }

  body.mesh.position.addScaledVector(rampNormal, body.radius - distance);
  const normalSpeed = body.velocity.dot(rampNormal);

  if (normalSpeed < 0) {
    body.velocity.addScaledVector(rampNormal, -(1.28 * normalSpeed));
  }

  const tangentVelocity = body.velocity.clone().addScaledVector(rampNormal, -body.velocity.dot(rampNormal));
  body.velocity.addScaledVector(tangentVelocity, -0.035);
  body.angularVelocity.add(new THREE.Vector3(z * 0.05, -x * 0.02, -x * 0.06));
  body.restingFrames = body.velocity.lengthSq() < 0.08 ? body.restingFrames + 1 : 0;
  return 1;
}

function resolveFloorContact(body: Body): number {
  const bottom = body.mesh.position.y - body.radius;

  if (bottom >= floor.position.y) {
    return 0;
  }

  body.mesh.position.y += floor.position.y - bottom;

  if (body.velocity.y < 0) {
    body.velocity.y *= -0.32;
    body.velocity.x *= 0.82;
    body.velocity.z *= 0.82;
  }

  return 1;
}

function resolveBodyContacts(): number {
  let contacts = 0;

  for (let i = 0; i < bodies.length; i += 1) {
    for (let j = i + 1; j < bodies.length; j += 1) {
      const a = bodies[i];
      const b = bodies[j];
      const delta = b.mesh.position.clone().sub(a.mesh.position);
      const distance = delta.length();
      const minDistance = a.radius + b.radius;

      if (distance <= 0.0001 || distance >= minDistance) {
        continue;
      }

      const normal = delta.multiplyScalar(1 / distance);
      const penetration = minDistance - distance;
      a.mesh.position.addScaledVector(normal, -penetration * 0.5);
      b.mesh.position.addScaledVector(normal, penetration * 0.5);

      const relative = b.velocity.clone().sub(a.velocity);
      const closingSpeed = relative.dot(normal);

      if (closingSpeed < 0) {
        const impulse = -(1.08 * closingSpeed) / 2;
        a.velocity.addScaledVector(normal, -impulse);
        b.velocity.addScaledVector(normal, impulse);
      }

      a.velocity.multiplyScalar(0.994);
      b.velocity.multiplyScalar(0.994);
      a.angularVelocity.add(new THREE.Vector3(normal.z, 0.2, -normal.x).multiplyScalar(0.12));
      b.angularVelocity.add(new THREE.Vector3(-normal.z, -0.2, normal.x).multiplyScalar(0.12));
      contacts += 1;
    }
  }

  return contacts;
}

let liveContacts = 0;

function stepPhysics(dt: number): void {
  liveContacts = 0;
  const gravity = new THREE.Vector3(0, -9.8, 0);

  for (const body of bodies) {
    body.velocity.addScaledVector(gravity, dt);

    if (body.restingFrames < 28) {
      body.mesh.position.addScaledVector(body.velocity, dt);
    } else {
      body.velocity.multiplyScalar(0.94);
    }

    body.mesh.rotation.x += body.angularVelocity.x * dt;
    body.mesh.rotation.y += body.angularVelocity.y * dt;
    body.mesh.rotation.z += body.angularVelocity.z * dt;
    body.angularVelocity.multiplyScalar(0.992);

    liveContacts += resolveRampContact(body);
    liveContacts += resolveFloorContact(body);
  }

  for (let i = 0; i < 3; i += 1) {
    liveContacts += resolveBodyContacts();
  }

  if (contactCountEl) {
    contactCountEl.textContent = liveContacts.toString();
  }
}

  controls.update();
  renderer.render(scene, camera);
}

}
