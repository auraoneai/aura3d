const COLS = 5;

const PITCH = 18; // distance between lot centres (leaves room for streets)
const HALF_W = ((COLS - 1) * PITCH) / 2;
const HALF_D = ((ROWS - 1) * PITCH) / 2;

// Deterministic pseudo-random so the layout is stable across runs.
let seed = 1337;
function rand(): number {
  seed = (seed * 1664525 + 1013904223) % 4294967296;
  return seed / 4294967296;
}

// Window texture — a grid of lit windows on a dark facade.
// Used as both colour map and emissive map so the windows glow at night.
// ---------------------------------------------------------------------------

function makeWindowTexture(): THREE.Texture {
  const cell = 32;
  const cols = 4;
  const rows = 8;
  const canvas = document.createElement('canvas');
  canvas.width = cols * cell;
  canvas.height = rows * cell;
  const ctx = canvas.getContext('2d')!;

  // Facade
  ctx.fillStyle = '#1b1d27';
  ctx.fillRect(0, 0, canvas.width, canvas.height);

  // Windows
  const pad = 6;
  for (let y = 0; y < rows; y++) {
    for (let x = 0; x < cols; x++) {
      // Vary window brightness for a believable lit-up look.
      const lit = rand();
      const shade =
        lit > 0.55
          ? '#fff2c2' // bright warm light
          : lit > 0.3
            ? '#cdd6ff' // cool light
            : '#2c2f3d'; // dark / empty
      ctx.fillStyle = shade;
      ctx.fillRect(x * cell + pad, y * cell + pad, cell - pad * 2, cell - pad * 2);
    }
  }

  const tex = new THREE.CanvasTexture(canvas);
  tex.wrapS = THREE.RepeatWrapping;
  tex.wrapT = THREE.RepeatWrapping;
  tex.colorSpace = THREE.SRGBColorSpace;
  return tex;
}

asphalt.rotation.x = -Math.PI / 2;
asphalt.receiveShadow = true;
groundGroup.add(asphalt);

const sidewalkMat = new THREE.MeshStandardMaterial({ color: 0x8a8a92, roughness: 1 });
const laneMat = new THREE.MeshStandardMaterial({

  emissive: 0x6b5a10,

const buildings: THREE.Mesh[] = [];
const buildingMaterials: THREE.MeshStandardMaterial[] = [];

const facadeColors = [0x6b7785, 0x7d6f63, 0x5f6b78, 0x8a8278, 0x6a6f77];

  for (let c = 0; c < COLS; c++) {
    const x = c * PITCH - HALF_W;
    const z = r * PITCH - HALF_D;

    // Sidewalk pad
    const pad = new THREE.Mesh(new THREE.BoxGeometry(13, 0.4, 13), sidewalkMat);
    pad.position.set(x, 0.2, z);
    pad.receiveShadow = true;
    groundGroup.add(pad);

    // Building — varying height
    const w = 6 + rand() * 4;
    const d = 6 + rand() * 4;
    const h = 8 + rand() * 34; // strongly varying heights

    const tex = baseWindowTexture.clone();
    tex.needsUpdate = true;
    tex.wrapS = tex.wrapT = THREE.RepeatWrapping;
    // Scale window grid to building size so windows look consistent.
    tex.repeat.set(Math.max(2, Math.round(w / 3)), Math.max(3, Math.round(h / 3)));

    const mat = new THREE.MeshStandardMaterial({
      color: facadeColors[(r * COLS + c) % facadeColors.length],
      map: tex,
      emissive: 0xffffff,
      emissiveMap: tex,
      emissiveIntensity: 0.0, // raised at night
      roughness: 0.7,
      metalness: 0.1,
    });

    const b = new THREE.Mesh(new THREE.BoxGeometry(w, h, d), mat);
    b.position.set(x, h / 2 + 0.4, z);
    b.castShadow = true;
    b.receiveShadow = true;
    groundGroup.add(b);
    buildings.push(b);
    buildingMaterials.push(mat);
  }
}

// Yellow centre lines down the vertical (north-south) streets between columns.
for (let c = 0; c < COLS - 1; c++) {

  const pole = new THREE.Mesh(new THREE.CylinderGeometry(0.25, 0.3, 9, 8), poleMat);

  light.position.set(1.9, 8.7, 0);
  group.add(light);

  group.position.set(x, 0.4, z);
  groundGroup.add(group);

  streetLights.push({ bulb: bulbMat, light });
}

for (let c = 0; c < COLS - 1; c++) {
  for (let r = 0; r < ROWS - 1; r++) {
