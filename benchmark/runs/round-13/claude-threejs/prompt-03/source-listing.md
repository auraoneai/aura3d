import * as THREE from 'three';
import { OrbitControls } from 'three/addons/controls/OrbitControls.js';
import { EffectComposer } from 'three/addons/postprocessing/EffectComposer.js';
import { RenderPass } from 'three/addons/postprocessing/RenderPass.js';
import { UnrealBloomPass } from 'three/addons/postprocessing/UnrealBloomPass.js';
import {
  CSS2DRenderer,
  CSS2DObject,
} from 'three/addons/renderers/CSS2DRenderer.js';

// ---------------------------------------------------------------------------
// Scene / camera / renderer
// ---------------------------------------------------------------------------
const container = document.getElementById('app') as HTMLDivElement;

const scene = new THREE.Scene();
scene.background = new THREE.Color(0x01010a);

const camera = new THREE.PerspectiveCamera(
  55,
  window.innerWidth / window.innerHeight,
  0.1,
  2000,
);
camera.position.set(0, 70, 130);

const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setPixelRatio(window.devicePixelRatio);
renderer.setSize(window.innerWidth, window.innerHeight);
renderer.toneMapping = THREE.ACESFilmicToneMapping;
container.appendChild(renderer.domElement);

// Label renderer (CSS2D) overlaid on top of the WebGL canvas.
const labelRenderer = new CSS2DRenderer();
labelRenderer.setSize(window.innerWidth, window.innerHeight);
labelRenderer.domElement.style.position = 'absolute';
labelRenderer.domElement.style.top = '0';
labelRenderer.domElement.style.left = '0';
labelRenderer.domElement.style.pointerEvents = 'none';
container.appendChild(labelRenderer.domElement);

// ---------------------------------------------------------------------------
// Lighting
// ---------------------------------------------------------------------------
// Ambient fill so the dark side of planets is not pure black.
scene.add(new THREE.AmbientLight(0xffffff, 0.18));
// Point light at the sun lights the planets.
const sunLight = new THREE.PointLight(0xfff4e0, 4, 0, 0.6);
scene.add(sunLight);

// Starfield backdrop.
function makeStars(count: number): THREE.Points {
  const positions = new Float32Array(count * 3);
  for (let i = 0; i < count; i++) {
    const r = 600 + Math.random() * 700;
    const theta = Math.random() * Math.PI * 2;
    const phi = Math.acos(2 * Math.random() - 1);
    positions[i * 3] = r * Math.sin(phi) * Math.cos(theta);
    positions[i * 3 + 1] = r * Math.sin(phi) * Math.sin(theta);
    positions[i * 3 + 2] = r * Math.cos(phi);
  }
  const geom = new THREE.BufferGeometry();
  geom.setAttribute('position', new THREE.BufferAttribute(positions, 3));
  const mat = new THREE.PointsMaterial({ color: 0xffffff, size: 1.4 });
  return new THREE.Points(geom, mat);
}
scene.add(makeStars(2500));

// ---------------------------------------------------------------------------
// Label helper
// ---------------------------------------------------------------------------
function makeLabel(text: string): CSS2DObject {
  const div = document.createElement('div');
  div.textContent = text;
  div.style.color = '#ffffff';
  div.style.fontFamily =
    'system-ui, -apple-system, Segoe UI, Roboto, sans-serif';
  div.style.fontSize = '13px';
  div.style.fontWeight = '600';
  div.style.letterSpacing = '0.5px';
  div.style.padding = '2px 6px';
  div.style.borderRadius = '4px';
  div.style.background = 'rgba(0, 0, 0, 0.55)';
  div.style.border = '1px solid rgba(255, 255, 255, 0.25)';
  div.style.whiteSpace = 'nowrap';
  div.style.textShadow = '0 1px 2px rgba(0,0,0,0.9)';
  return new CSS2DObject(div);
}

// ---------------------------------------------------------------------------
// Sun (emissive + glow sprite). Bloom amplifies the brightness.
// ---------------------------------------------------------------------------
const SUN_RADIUS = 8;
const sun = new THREE.Mesh(
  new THREE.SphereGeometry(SUN_RADIUS, 64, 64),
  new THREE.MeshBasicMaterial({ color: 0xffcc44 }),
);
scene.add(sun);

const sunLabel = makeLabel('Sun');
sunLabel.position.set(0, SUN_RADIUS + 4, 0);
sun.add(sunLabel);

// Soft additive glow halo around the sun (helps the bloom read as a corona).
function makeGlowTexture(): THREE.Texture {
  const size = 256;
  const canvas = document.createElement('canvas');
  canvas.width = size;
  canvas.height = size;
  const ctx = canvas.getContext('2d')!;
  const grad = ctx.createRadialGradient(
    size / 2,
    size / 2,
    0,
    size / 2,
    size / 2,
    size / 2,
  );
  grad.addColorStop(0, 'rgba(255,230,150,1)');
  grad.addColorStop(0.25, 'rgba(255,180,60,0.7)');
  grad.addColorStop(1, 'rgba(255,140,0,0)');
  ctx.fillStyle = grad;
  ctx.fillRect(0, 0, size, size);
  const tex = new THREE.CanvasTexture(canvas);
  tex.colorSpace = THREE.SRGBColorSpace;
  return tex;
}
const glow = new THREE.Sprite(
  new THREE.SpriteMaterial({
    map: makeGlowTexture(),
    color: 0xffffff,
    transparent: true,
    blending: THREE.AdditiveBlending,
    depthWrite: false,
  }),
);
glow.scale.setScalar(SUN_RADIUS * 6);
sun.add(glow);

// ---------------------------------------------------------------------------
// Planets
// ---------------------------------------------------------------------------
interface PlanetDef {
  name: string;
  radius: number;
  distance: number;
  speed: number; // radians per second
  color: number;
}

const PLANETS: PlanetDef[] = [
  { name: 'Mercury', radius: 1.2, distance: 18, speed: 0.9, color: 0xa9a9a9 },
  { name: 'Venus', radius: 2.0, distance: 28, speed: 0.65, color: 0xe7c07b },
  { name: 'Earth', radius: 2.2, distance: 40, speed: 0.5, color: 0x3a7bd5 },
  { name: 'Mars', radius: 1.6, distance: 54, speed: 0.4, color: 0xc1440e },
  { name: 'Jupiter', radius: 4.5, distance: 74, speed: 0.24, color: 0xd8a47f },
  { name: 'Saturn', radius: 3.8, distance: 96, speed: 0.16, color: 0xead6a4 },
];

interface Planet {
  pivot: THREE.Object3D;
  mesh: THREE.Mesh;
  speed: number;
}

const planets: Planet[] = [];

for (const def of PLANETS) {
  // Pivot rotates about the sun; the mesh sits at the orbital distance.
  const pivot = new THREE.Object3D();
  scene.add(pivot);

  const mesh = new THREE.Mesh(
    new THREE.SphereGeometry(def.radius, 48, 48),
    new THREE.MeshStandardMaterial({
      color: def.color,
      roughness: 0.85,
      metalness: 0.05,
    }),
  );
  mesh.position.x = def.distance;
  pivot.add(mesh);

  // Visible orbit ring at the planet's distance.
  const orbit = new THREE.Mesh(
    new THREE.RingGeometry(def.distance - 0.08, def.distance + 0.08, 128),
    new THREE.MeshBasicMaterial({
      color: 0x4466aa,
      side: THREE.DoubleSide,
      transparent: true,
      opacity: 0.35,
    }),
  );
  orbit.rotation.x = -Math.PI / 2;
  scene.add(orbit);

  // Decorative ring for Saturn.
  if (def.name === 'Saturn') {
    const ring = new THREE.Mesh(
      new THREE.RingGeometry(def.radius + 1.2, def.radius + 3.2, 64),
      new THREE.MeshBasicMaterial({
        color: 0xd9c08a,
        side: THREE.DoubleSide,
        transparent: true,
        opacity: 0.7,
      }),
    );
    ring.rotation.x = -Math.PI / 2.2;
    mesh.add(ring);
  }

  const label = makeLabel(def.name);
  label.position.set(0, def.radius + 2.5, 0);
  mesh.add(label);

  planets.push({ pivot, mesh, speed: def.speed });
}

// ---------------------------------------------------------------------------
// Controls (orbit camera)
// ---------------------------------------------------------------------------
const controls = new OrbitControls(camera, renderer.domElement);
controls.enableDamping = true;
controls.dampingFactor = 0.05;
controls.minDistance = 30;
controls.maxDistance = 400;
controls.target.set(0, 0, 0);
controls.autoRotate = true;
controls.autoRotateSpeed = 0.3;
controls.update();

// ---------------------------------------------------------------------------
// Post-processing: bloom on the sun
// ---------------------------------------------------------------------------
const composer = new EffectComposer(renderer);
composer.addPass(new RenderPass(scene, camera));
const bloomPass = new UnrealBloomPass(
  new THREE.Vector2(window.innerWidth, window.innerHeight),
  1.4, // strength
  0.6, // radius
  0.85, // threshold — only the bright sun blooms
);
composer.addPass(bloomPass);

// ---------------------------------------------------------------------------
// Resize
// ---------------------------------------------------------------------------
window.addEventListener('resize', () => {
  const w = window.innerWidth;
  const h = window.innerHeight;
  camera.aspect = w / h;
  camera.updateProjectionMatrix();
  renderer.setSize(w, h);
  composer.setSize(w, h);
  labelRenderer.setSize(w, h);
});

// ---------------------------------------------------------------------------
// Animation loop
// ---------------------------------------------------------------------------
const clock = new THREE.Clock();

function animate() {
  const dt = clock.getDelta();
  const t = clock.elapsedTime;

  for (const p of planets) {
    p.pivot.rotation.y = t * p.speed;
    p.mesh.rotation.y += dt * 0.5;
  }
  sun.rotation.y += dt * 0.1;

  controls.update();
  composer.render();
  labelRenderer.render(scene, camera);
}

renderer.setAnimationLoop(animate);
