## src/main.ts

import * as THREE from 'three'

/* ------------------------------------------------------------------ *
 * Mini-Golf Hole
 * - Flat green with raised borders
 * - One obstacle (a brick wall block)
 * - A ball with simple rolling physics (friction + wall/obstacle bounces)
 * - Click & drag from the ball to aim, release to shoot (slingshot)
 * - Stroke counter (HUD)
 * - Camera follows the ball
 * ------------------------------------------------------------------ */

// ----- Course constants (XZ plane, Y up) -----
const GREEN_HALF_X = 11
const GREEN_HALF_Z = 17
const WALL_HEIGHT = 1.2
const WALL_THICK = 0.6

const BALL_RADIUS = 0.55
const TEE = new THREE.Vector3(0, BALL_RADIUS, GREEN_HALF_Z - 3)
const HOLE = new THREE.Vector3(0, 0, -(GREEN_HALF_Z - 3))
const HOLE_RADIUS = 1.05

const MAX_SHOT_SPEED = 24 // units / second
const FRICTION = 1.05 // higher = stops sooner
const STILL_SPEED = 0.18 // below this the ball is considered stopped
const RESTITUTION = 0.62 // energy kept on bounce

// Obstacle (axis-aligned box) — the single required obstacle.
const OBSTACLE_CENTER = new THREE.Vector3(-1.5, 0, 1.5)
const OBSTACLE_HALF = new THREE.Vector3(4, 0.9, 0.9)

// ----- Renderer -----
const app = document.getElementById('app') as HTMLDivElement
const renderer = new THREE.WebGLRenderer({ antialias: true })
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2))
renderer.setSize(window.innerWidth, window.innerHeight)
renderer.shadowMap.enabled = true
renderer.shadowMap.type = THREE.PCFSoftShadowMap
renderer.outputColorSpace = THREE.SRGBColorSpace
app.appendChild(renderer.domElement)

// ----- Scene & camera -----
const scene = new THREE.Scene()
scene.background = new THREE.Color(0x0b132b)
scene.fog = new THREE.Fog(0x0b132b, 45, 95)

const camera = new THREE.PerspectiveCamera(
  55,
  window.innerWidth / window.innerHeight,
  0.1,
  500,
)
camera.position.set(0, 14, TEE.z + 16)
camera.lookAt(0, 0, 0)

// ----- Lights -----
scene.add(new THREE.HemisphereLight(0xbfd4ff, 0x20301a, 0.9))
const sun = new THREE.DirectionalLight(0xffffff, 1.8)
sun.position.set(14, 26, 12)
sun.castShadow = true
sun.shadow.mapSize.set(2048, 2048)
sun.shadow.camera.near = 1
sun.shadow.camera.far = 90
sun.shadow.camera.left = -32
sun.shadow.camera.right = 32
sun.shadow.camera.top = 32
sun.shadow.camera.bottom = -32
sun.shadow.bias = -0.0004
scene.add(sun)

// ----- The green (with mowed-lawn stripes) -----
function makeGreenTexture(): THREE.Texture {
  const size = 512
  const c = document.createElement('canvas')
  c.width = c.height = size
  const ctx = c.getContext('2d')!
  ctx.fillStyle = '#2f9e44'
  ctx.fillRect(0, 0, size, size)
  const stripes = 10
  for (let i = 0; i < stripes; i++) {
    ctx.fillStyle = i % 2 === 0 ? '#37b24d' : '#2b8a3e'
    ctx.fillRect(0, (i * size) / stripes, size, size / stripes)
  }
  const tex = new THREE.CanvasTexture(c)
  tex.colorSpace = THREE.SRGBColorSpace
  tex.wrapS = tex.wrapT = THREE.RepeatWrapping
  tex.repeat.set(1, 1)
  tex.anisotropy = renderer.capabilities.getMaxAnisotropy()
  return tex
}

const greenGroup = new THREE.Group()
scene.add(greenGroup)

const green = new THREE.Mesh(
  new THREE.BoxGeometry(GREEN_HALF_X * 2, 1, GREEN_HALF_Z * 2),
  new THREE.MeshStandardMaterial({ map: makeGreenTexture(), roughness: 0.95, metalness: 0 }),
)
green.position.y = -0.5
green.receiveShadow = true
greenGroup.add(green)

// Raised borders so the ball stays on the green.
const wallMat = new THREE.MeshStandardMaterial({ color: 0x6f4e37, roughness: 0.8 })
function addWall(w: number, d: number, x: number, z: number) {
  const wall = new THREE.Mesh(new THREE.BoxGeometry(w, WALL_HEIGHT, d), wallMat)
  wall.position.set(x, WALL_HEIGHT / 2, z)
  wall.castShadow = true
  wall.receiveShadow = true
  greenGroup.add(wall)
}
const span = GREEN_HALF_X * 2 + WALL_THICK * 2
addWall(span, WALL_THICK, 0, -GREEN_HALF_Z - WALL_THICK / 2)
addWall(span, WALL_THICK, 0, GREEN_HALF_Z + WALL_THICK / 2)
addWall(WALL_THICK, GREEN_HALF_Z * 2, -GREEN_HALF_X - WALL_THICK / 2, 0)
addWall(WALL_THICK, GREEN_HALF_Z * 2, GREEN_HALF_X + WALL_THICK / 2, 0)

// ----- The obstacle -----
const obstacle = new THREE.Mesh(
  new THREE.BoxGeometry(OBSTACLE_HALF.x * 2, OBSTACLE_HALF.y * 2, OBSTACLE_HALF.z * 2),
  new THREE.MeshStandardMaterial({ color: 0xc0392b, roughness: 0.7 }),
)
obstacle.position.set(OBSTACLE_CENTER.x, OBSTACLE_HALF.y, OBSTACLE_CENTER.z)
obstacle.castShadow = true
obstacle.receiveShadow = true
scene.add(obstacle)

// ----- The hole + flag -----
const hole = new THREE.Mesh(
  new THREE.CylinderGeometry(HOLE_RADIUS, HOLE_RADIUS, 0.5, 32),
  new THREE.MeshStandardMaterial({ color: 0x05140a, roughness: 1 }),
)
hole.position.set(HOLE.x, -0.25, HOLE.z)
hole.receiveShadow = true
scene.add(hole)

const ringMat = new THREE.MeshStandardMaterial({ color: 0xf1f3f5, roughness: 0.6 })
const ring = new THREE.Mesh(new THREE.TorusGeometry(HOLE_RADIUS, 0.07, 12, 40), ringMat)
ring.rotation.x = -Math.PI / 2
ring.position.set(HOLE.x, 0.02, HOLE.z)
scene.add(ring)

const flag = new THREE.Group()
const pole = new THREE.Mesh(
  new THREE.CylinderGeometry(0.05, 0.05, 3.2, 8),
  new THREE.MeshStandardMaterial({ color: 0xf8f9fa }),
)
pole.position.y = 1.6
pole.castShadow = true
flag.add(pole)
const flagCloth = new THREE.Mesh(
  new THREE.PlaneGeometry(1.1, 0.7),
  new THREE.MeshStandardMaterial({ color: 0xffd43b, side: THREE.DoubleSide }),
)
flagCloth.position.set(0.55, 2.9, 0)
flag.add(flagCloth)
flag.position.set(HOLE.x, 0, HOLE.z)
scene.add(flag)

// ----- The ball -----
const ball = new THREE.Mesh(
  new THREE.SphereGeometry(BALL_RADIUS, 32, 32),
  new THREE.MeshStandardMaterial({ color: 0xffffff, roughness: 0.35, metalness: 0.05 }),
)
ball.castShadow = true
ball.position.copy(TEE)
scene.add(ball)

const velocity = new THREE.Vector3()
let sunk = false

// ----- Aiming arrow -----
const arrow = new THREE.ArrowHelper(
  new THREE.Vector3(0, 0, -1),
  ball.position,
  0,
  0x111111,
  0.9,
  0.6,
)
arrow.visible = false
scene.add(arrow)

// ----- HUD -----
const scoreEl = document.getElementById('score') as HTMLDivElement
const statusEl = document.getElementById('status') as HTMLDivElement
let strokes = 0
function renderHud() {
  scoreEl.innerHTML = `Strokes: ${strokes}` + (sunk ? '' : ' <small>(par 3)</small>')
}
renderHud()

// ----- Input: click-drag to aim, release to shoot -----
const raycaster = new THREE.Raycaster()
const pointer = new THREE.Vector2()
const groundPlane = new THREE.Plane(new THREE.Vector3(0, 1, 0), 0)
const hitPoint = new THREE.Vector3()
const shootDir = new THREE.Vector3()

let aiming = false
let power = 0 // 0..1

function pointerToGround(ev: PointerEvent): boolean {
  pointer.x = (ev.clientX / window.innerWidth) * 2 - 1
  pointer.y = -(ev.clientY / window.innerHeight) * 2 + 1
  raycaster.setFromCamera(pointer, camera)
  return raycaster.ray.intersectPlane(groundPlane, hitPoint) !== null
}

function isStill(): boolean {
  return velocity.length() < STILL_SPEED
}

function updateAim() {
  // Slingshot: pull back from the ball; the ball travels the opposite way.
  shootDir.set(ball.position.x - hitPoint.x, 0, ball.position.z - hitPoint.z)
  const drag = shootDir.length()
  if (drag < 0.001) {
    power = 0
    arrow.visible = false
    return
  }
  shootDir.normalize()
  power = Math.min(drag / 11, 1)
  arrow.visible = power > 0.02
  arrow.position.set(ball.position.x, BALL_RADIUS, ball.position.z)
  arrow.setDirection(shootDir)
  arrow.setLength(1.2 + power * 7, 0.9, 0.6)
  // green -> yellow -> red as power grows
  const col = new THREE.Color().setHSL((1 - power) * 0.33, 0.9, 0.5)
  arrow.setColor(col)
}

renderer.domElement.addEventListener('pointerdown', (ev) => {
  if (sunk || !isStill()) return
  if (!pointerToGround(ev)) return
  aiming = true
  updateAim()
})

renderer.domElement.addEventListener('pointermove', (ev) => {
  if (!aiming) return
  if (pointerToGround(ev)) updateAim()
})

function endAim(shoot: boolean) {
  if (!aiming) return
  aiming = false
  arrow.visible = false
  if (shoot && power > 0.04) {
    velocity.copy(shootDir).multiplyScalar(power * MAX_SHOT_SPEED)
    strokes++
    statusEl.textContent = ''
    renderHud()
  }
  power = 0
}

renderer.domElement.addEventListener('pointerup', () => endAim(true))
renderer.domElement.addEventListener('pointerleave', () => endAim(false))

window.addEventListener('keydown', (ev) => {
  if (ev.key === 'r' || ev.key === 'R') resetGame()
})

function resetGame() {
  ball.position.copy(TEE)
  ball.scale.setScalar(1)
  velocity.set(0, 0, 0)
  sunk = false
  strokes = 0
  statusEl.textContent = ''
  renderHud()
}

// ----- Physics step -----
const tmp = new THREE.Vector3()
const rollAxis = new THREE.Vector3()
const up = new THREE.Vector3(0, 1, 0)

function stepPhysics(dt: number) {
  if (sunk) {
    // Drop into the cup and shrink slightly for a "sink" effect.
    if (ball.position.y > -0.4) {
      ball.position.y -= dt * 3
      ball.scale.multiplyScalar(1 - dt * 1.5)
    }
    return
  }

  const speed = velocity.length()
  if (speed <= 0) return

  // Friction (frame-rate independent).
  const decay = Math.max(0, 1 - FRICTION * dt)
  velocity.multiplyScalar(decay)
  if (velocity.length() < STILL_SPEED * 0.4) {
    velocity.set(0, 0, 0)
    return
  }

  // Integrate.
  ball.position.x += velocity.x * dt
  ball.position.z += velocity.z * dt
  ball.position.y = BALL_RADIUS

  // Rolling visual.
  const moved = tmp.set(velocity.x, 0, velocity.z)
  const movedLen = moved.length()
  if (movedLen > 1e-4) {
    rollAxis.crossVectors(up, moved.normalize()).normalize()
    ball.rotateOnWorldAxis(rollAxis, (movedLen * dt) / BALL_RADIUS)
  }

  collideWalls()
  collideObstacle()
  checkHole()
}

function collideWalls() {
  const limX = GREEN_HALF_X - BALL_RADIUS
  const limZ = GREEN_HALF_Z - BALL_RADIUS
  if (ball.position.x > limX) {
    ball.position.x = limX
    velocity.x = -Math.abs(velocity.x) * RESTITUTION
  } else if (ball.position.x < -limX) {
    ball.position.x = -limX
    velocity.x = Math.abs(velocity.x) * RESTITUTION
  }
  if (ball.position.z > limZ) {
    ball.position.z = limZ
    velocity.z = -Math.abs(velocity.z) * RESTITUTION
  } else if (ball.position.z < -limZ) {
    ball.position.z = -limZ
    velocity.z = Math.abs(velocity.z) * RESTITUTION
  }
}

function collideObstacle() {
  const minX = OBSTACLE_CENTER.x - OBSTACLE_HALF.x - BALL_RADIUS
  const maxX = OBSTACLE_CENTER.x + OBSTACLE_HALF.x + BALL_RADIUS
  const minZ = OBSTACLE_CENTER.z - OBSTACLE_HALF.z - BALL_RADIUS
  const maxZ = OBSTACLE_CENTER.z + OBSTACLE_HALF.z + BALL_RADIUS
  const p = ball.position
  if (p.x > minX && p.x < maxX && p.z > minZ && p.z < maxZ) {
    // Resolve along the axis of least penetration.
    const penLeft = p.x - minX
    const penRight = maxX - p.x
    const penDown = p.z - minZ
    const penUp = maxZ - p.z
    const minPen = Math.min(penLeft, penRight, penDown, penUp)
    if (minPen === penLeft) {
      p.x = minX
      velocity.x = -Math.abs(velocity.x) * RESTITUTION
    } else if (minPen === penRight) {
      p.x = maxX
      velocity.x = Math.abs(velocity.x) * RESTITUTION
    } else if (minPen === penDown) {
      p.z = minZ
      velocity.z = -Math.abs(velocity.z) * RESTITUTION
    } else {
      p.z = maxZ
      velocity.z = Math.abs(velocity.z) * RESTITUTION
    }
  }
}

function checkHole() {
  const dx = ball.position.x - HOLE.x
  const dz = ball.position.z - HOLE.z
  const dist = Math.hypot(dx, dz)
  // Must be near the cup and not flying too fast to drop in.
  if (dist < HOLE_RADIUS - 0.15 && velocity.length() < MAX_SHOT_SPEED * 0.55) {
    sunk = true
    velocity.set(0, 0, 0)
    ball.position.x = HOLE.x
    ball.position.z = HOLE.z
    statusEl.textContent = `🏌️  In the hole in ${strokes}! Press R to play again.`
    renderHud()
  }
}

// ----- Follow camera -----
const camOffset = new THREE.Vector3(0, 11, 14)
const desired = new THREE.Vector3()
const lookTarget = new THREE.Vector3()

function updateCamera(dt: number) {
  desired.copy(ball.position).add(camOffset)
  // Smoothly trail the ball.
  const k = 1 - Math.pow(0.0015, dt)
  camera.position.lerp(desired, k)
  lookTarget.lerp(ball.position, 1 - Math.pow(0.0008, dt))
  camera.lookAt(lookTarget)
}
lookTarget.copy(ball.position)

// ----- Loop -----
const clock = new THREE.Clock()
function animate() {
  const dt = Math.min(clock.getDelta(), 0.05)
  stepPhysics(dt)
  updateCamera(dt)
  // Gentle flag wave.
  flagCloth.rotation.y = Math.sin(clock.elapsedTime * 2) * 0.25
  renderer.render(scene, camera)
}
renderer.setAnimationLoop(animate)

// ----- Resize -----
window.addEventListener('resize', () => {
  camera.aspect = window.innerWidth / window.innerHeight
  camera.updateProjectionMatrix()
  renderer.setSize(window.innerWidth, window.innerHeight)
})
