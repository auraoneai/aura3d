  const dt = Math.min(deltaSeconds, 1 / 30);
  let contacts = 0;
  const contactPoints: THREE.Vector3[] = [];

      body.angularVelocity.add(new THREE.Vector3(0.02, 0.06, 0.015));
      body.restingFrames += Math.abs(normalSpeed) < 0.12 ? 1 : 0;
      contactPoints.push(body.position.clone().addScaledVector(rampNormal, -radius));
    }

    if (body.position.y < -1.45) {
      contacts += 1;
      body.position.y = -1.45;
      if (body.velocity.y < 0) {
        body.velocity.y *= -0.36;

        body.velocity.z *= 0.72;

      scratchA.copy(b.position).sub(a.position);
      const distance = scratchA.length();

      if (distance > 0.0001 && distance < minimumDistance) {

        const normal = scratchA.multiplyScalar(1 / distance);

        b.position.addScaledVector(normal, overlap * 0.5);

          b.velocity.addScaledVector(normal, impulse);

        b.angularVelocity.addScaledVector(normal, 0.22);
        contactPoints.push(a.position.clone().lerp(b.position, 0.5));
      }
    }

    body.mesh.position.copy(body.position);

    body.mesh.rotateZ(body.angularVelocity.z * dt);

}
