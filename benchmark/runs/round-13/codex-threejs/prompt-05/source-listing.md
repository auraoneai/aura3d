function updateHover(): void {
  raycaster.setFromCamera(pointer, camera);
  const intersections = raycaster.intersectObjects(bars.map((bar) => bar.mesh), false);
  const nextHover = intersections.length
    ? bars.find((bar) => bar.mesh === intersections[0].object) ?? null
    : null;

  if (hoveredBar && hoveredBar !== nextHover) {
    hoveredBar.mesh.material.emissive.setHex(0x000000);
    hoveredBar.mesh.material.opacity = 1;
    hoveredBar.mesh.scale.x = 1;
    hoveredBar.mesh.scale.z = 1;
  }

  if (nextHover) {
    nextHover.mesh.material.emissive.setHex(0xfff2b0);
    nextHover.mesh.scale.x = 1.12;
    nextHover.mesh.scale.z = 1.12;
  }

  hoveredBar = nextHover;
  renderer.domElement.style.cursor = hoveredBar ? 'pointer' : 'grab';
}

function animate(): void {
  const elapsed = clock.getElapsedTime();

  for (const [index, bar] of bars.entries()) {
    const wave = Math.sin(elapsed * 0.9 + bar.phase + index * 0.09);
    const height = THREE.MathUtils.clamp(bar.base + wave * bar.amplitude, minHeight, maxHeight);
    bar.mesh.scale.y = height;
    bar.mesh.position.y = height / 2;
    bar.mesh.material.color.copy(heightToColor(height));

  requestAnimationFrame(animate);
}

}

}
