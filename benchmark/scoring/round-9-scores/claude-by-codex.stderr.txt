// Mini-Golf Hole — raw Three.js (0.165.0)
// A flat green, one obstacle, a ball with simple physics, click-to-aim-and-shoot,
// a stroke/score counter, and a camera that follows the ball.

import * as THREE from 'three';

// ---------------------------------------------------------------------------
// Constants / course layout
// ---------------------------------------------------------------------------
const GREEN_HALF_X = 9;   // green spans [-9, 9] on X
const GREEN_HALF_Z = 16;  // green spans [-16, 16] on Z
const WALL_H = 0.9;
const BALL_R = 0.45;
const HOLE_R = 0.85;
const FRICTION = 1.6;        // velocity damping per second (exponential-ish)
const STOP_SPEED = 0.35;     // below this the ball is considered at rest
const MAX_POWER = 26;        // maximum launch speed
const RESTITUTION = 0.72;    // bounce energy retained on wall/obstacle hits

const START_POS = new THREE.Vector3(0, BALL_R, GREEN_HALF_Z - 3);
const HOLE_POS = new THREE.Vector3(0, 0, -GREEN_HALF_Z + 4);

// ---------------------------------------------------------------------------
// Renderer / scene / camera
// ---------------------------------------------------------------------------
const app = document.getElementById('app') as HTMLDivElement;

const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
renderer.setSize(window.innerWidth, window.innerHeight);
renderer.shadowMap.enabled = true;
renderer.shadowMap.type = THREE.PCFSoftShadowMap;
renderer.outputColorSpace = THREE.SRGBColorSpace;
app.appendChild(renderer.domElement);

const scene = new THREE.Scene();
scene.background = new THREE.Color(0x8ec6ff);
scene.fog = new THREE.Fog(0x8ec6ff, 40, 90);

const camera = new THREE.PerspectiveCamera(
  55,
  window.innerWidth / window.innerHeight,
  0.1,
  500,
);
camera.position.set(0, 14, GREEN_HALF_Z + 10);
camera.lookAt(START_POS);

// ---------------------------------------------------------------------------
// Lighting
// ---------------------------------------------------------------------------
const hemi = new THREE.HemisphereLight(0xffffff, 0x335522, 0.9);
scene.add(hemi);

const sun = new THREE.DirectionalLight(0xffffff, 2.0);
sun.position.set(-12, 24, 14);
sun.castShadow = true;
sun.shadow.mapSize.set(2048, 2048);
sun.shadow.camera.near = 1;
sun.shadow.camera.far = 80;
sun.shadow.camera.left = -28;
sun.shadow.camera.right = 28;
sun.shadow.camera.top = 32;
sun.shadow.camera.bottom = -32;
scene.add(sun);

// ---------------------------------------------------------------------------
// The green (flat playing surface)
// ---------------------------------------------------------------------------
const greenMat = new THREE.MeshStandardMaterial({ color: 0x2f9e44, roughness: 0.95 });
const green = new THREE.Mesh(
  new THREE.PlaneGeometry(GREEN_HALF_X * 2, GREEN_HALF_Z * 2),
  greenMat,
);
green.rotation.x = -Math.PI / 2;
green.receiveShadow = true;
scene.add(green);

// Subtle mowing stripes so the green reads clearly as a golf surface.
const stripeMat = new THREE.MeshStandardMaterial({ color: 0x37b24d, roughness: 0.95 });
for (let i = -GREEN_HALF_Z + 2; i < GREEN_HALF_Z; i += 4) {
  const stripe = new THREE.Mesh(
    new THREE.PlaneGeometry(GREEN_HALF_X * 2, 2),
    stripeMat,
  );
  stripe.rotation.x = -Math.PI / 2;
  stripe.position.set(0, 0.01, i);
  stripe.receiveShadow = true;
  scene.add(stripe);
}

// Wooden border walls keep the ball on the green and provide bounce surfaces.
const wallMat = new THREE.MeshStandardMaterial({ color: 0x8a5a2b, roughness: 0.8 });
function addWall(w: number, d: number, x: number, z: number): void {
  const wall = new THREE.Mesh(new THREE.BoxGeometry(w, WALL_H, d), wallMat);
  wall.position.set(x, WALL_H / 2, z);
  wall.castShadow = true;
  wall.receiveShadow = true;
  scene.add(wall);
}
const T = 0.6; // wall thickness
addWall(GREEN_HALF_X * 2 + T * 2, T, 0, -GREEN_HALF_Z - T / 2); // far
addWall(GREEN_HALF_X * 2 + T * 2, T, 0, GREEN_HALF_Z + T / 2);  // near
addWall(T, GREEN_HALF_Z * 2 + T * 2, -GREEN_HALF_X - T / 2, 0); // left
addWall(T, GREEN_HALF_Z * 2 + T * 2, GREEN_HALF_X + T / 2, 0);  // right

import { camera, createAuraApp, effects, lights, prefabs, scene } from "@aura3d/engine";
import { ui } from "@aura3d/engine";
import "./style.css";

type TimeOfDay = "day" | "night";

// Procedural city block: 20 box buildings of varying heights with windows,
// streets, and street lights. The day/night toggle rebuilds the scene so the
// sky (background) and lighting visibly change between states.
function buildScene(timeOfDay: TimeOfDay) {
  const isNight = timeOfDay === "night";
  return scene()
    .background(isNight ? "#061018" : "#9ec9f0")
    .addMany(prefabs.cityBlock({ blocks: 20, litWindows: true, timeOfDay }))
    .add(effects.fog({ density: isNight ? 0.035 : 0.014 }))
    .add(effects.bloom({ intensity: isNight ? 0.16 : 0.05 }))
    .add(lights.ambient({
      intensity: isNight ? 0.28 : 0.92,
      color: isNight ? "#22304a" : "#dfeeff"
    }))
    .add(lights.directional({
      position: [6, 9, 4],
      intensity: isNight ? 0.18 : 1.45,
      color: isNight ? "#3a4b6e" : "#fff4e0"
    }))
    .add(lights.studio({ intensity: isNight ? 0.55 : 1.1 }))
    .camera(camera.perspective({ position: [0.6, 5.2, 9.2], target: [0, 0.42, -0.4], fov: 58 }));
}

ui.html("#app", `
  <div id="stage"></div>
  <button class="toggle" type="button" aria-pressed="true">night mode active — switch to day</button>
`);

const stage = ui.root("#stage");

let isNight = true;
let app = createAuraApp(stage, { scene: buildScene("night") });

ui.onClick(".toggle", (button) => {
  isNight = !isNight;
  app.dispose();
  stage.replaceChildren();
  app = createAuraApp(stage, { scene: buildScene(isNight ? "night" : "day") });
  ui.setText(button, isNight ? "night mode active — switch to day" : "day mode active — switch to night");
  ui.setPressed(button, isNight);
});

// Returns a pivot Group (placed at the joint) containing a limb built from a
// box "upper" segment and a box "lower" segment, with an optional foot.
// The lower segment lives in its own nested pivot so knees/elbows can flex.
interface Limb {
  pivot: THREE.Group; // shoulder / hip — primary swing
  lower: THREE.Group; // knee / elbow — secondary flex
}

function makeLimb(
  jointX: number,
  jointY: number,
  upperLen: number,
  lowerLen: number,
  thickness: number,
  mat: THREE.Material,
  withFoot: boolean,
): Limb {
  const pivot = new THREE.Group();
  pivot.position.set(jointX, jointY, 0);

  const upper = mesh(new THREE.BoxGeometry(thickness, upperLen, thickness), mat);
  upper.position.y = -upperLen / 2;
  pivot.add(upper);

  const lower = new THREE.Group();
  lower.position.y = -upperLen; // hinge at the end of the upper segment
  pivot.add(lower);

  const lowerSeg = mesh(
    new THREE.BoxGeometry(thickness * 0.9, lowerLen, thickness * 0.9),
    mat,
  );
  lowerSeg.position.y = -lowerLen / 2;
  lower.add(lowerSeg);

  if (withFoot) {
    const foot = mesh(new THREE.BoxGeometry(thickness * 1.1, 0.16, thickness * 1.9), shoeMat);
    foot.position.set(0, -lowerLen - 0.04, thickness * 0.45);
    lower.add(foot);
  } else {
    // hand
    const hand = mesh(new THREE.SphereGeometry(thickness * 0.62, 12, 12), skinMat);
    hand.position.y = -lowerLen - 0.02;
    lower.add(hand);
  }

  return { pivot, lower };
}

// (legs split, arms swinging) rather than a static neutral stance.

walker.position.x = -PATH_HALF;

const clock = new THREE.Clock();

function animate(): void {
  const t = clock.getElapsedTime();
  const phase = t * STRIDE_HZ * Math.PI * 2 + PHASE_OFFSET;

  // Forward travel with wrap-around so the figure keeps crossing the plane.
  walker.position.x = ((t * WALK_SPEED + PATH_HALF) % (PATH_HALF * 2)) - PATH_HALF;

  // Vertical bob: body rises twice per stride (once per footfall).
  walker.position.y = Math.abs(Math.sin(phase)) * 0.12;

  // Leg swing (opposite phase) + knee flex on the forward swing.
  const legSwing = 0.85;
  leftLeg.pivot.rotation.x = Math.sin(phase) * legSwing;
  rightLeg.pivot.rotation.x = Math.sin(phase + Math.PI) * legSwing;
  // Knees bend (always positive — knees only flex backward) as each leg lifts.
  leftLeg.lower.rotation.x = Math.max(0, -Math.sin(phase + 0.5)) * 1.1;
  rightLeg.lower.rotation.x = Math.max(0, -Math.sin(phase + Math.PI + 0.5)) * 1.1;

  // Arms swing opposite to the legs on the same side; elbows flex slightly.
  const armSwing = 0.7;
  leftArm.pivot.rotation.x = Math.sin(phase + Math.PI) * armSwing;
  rightArm.pivot.rotation.x = Math.sin(phase) * armSwing;
  leftArm.lower.rotation.x = (0.4 + 0.3 * Math.sin(phase + Math.PI)) * 0.9;
  rightArm.lower.rotation.x = (0.4 + 0.3 * Math.sin(phase)) * 0.9;

  body.rotation.z = Math.sin(phase) * 0.05;
  body.position.y = Math.sin(phase * 2) * 0.03;
